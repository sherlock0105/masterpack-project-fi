FORM make_hierarchy .

  IF gcl_tree IS BOUND.
    gcl_tree->delete_all_nodes( ).
  ENDIF.

  DATA: lt_data_tmp LIKE TABLE OF gs_data,
        lt_sum      LIKE TABLE OF gs_data, " 합계 보관용 테이블
        ls_sum      LIKE gs_data,
        lv_all_key  TYPE lvc_nkey,
        lv_cls_key  TYPE lvc_nkey,
        lv_node_key TYPE lvc_nkey.

  lt_data_tmp = gt_data.
  REFRESH: gt_make_tree, lt_sum.

  " 자산분류별 합계 집계
  LOOP AT lt_data_tmp ASSIGNING FIELD-SYMBOL(<fs_data_tmp>).
    CLEAR ls_sum.
    ls_sum-anlkl  = <fs_data_tmp>-anlkl.
    ls_sum-aprice = <fs_data_tmp>-aprice.
    ls_sum-accum  = <fs_data_tmp>-accum.
    ls_sum-book   = <fs_data_tmp>-book.
    ls_sum-waers  = <fs_data_tmp>-waers.
    COLLECT ls_sum INTO lt_sum.
  ENDLOOP.

  SORT lt_sum      BY anlkl.
  SORT lt_data_tmp BY anlkl anln1.

  " 전체(Root) 노드 생성 LV 0.
  PERFORM add_node USING ls_sum '전체' space lv_all_key.
  gv_all_node = lv_all_key.

  LOOP AT lt_sum INTO ls_sum.
    " LV 1 자산 분류 폴더
    PERFORM add_node USING ls_sum ls_sum-anlkl lv_all_key lv_cls_key.

    " LV 2 리프(자식) 노드
    LOOP AT lt_data_tmp INTO gs_data WHERE anlkl = ls_sum-anlkl.
      PERFORM add_node USING gs_data gs_data-anlnt lv_cls_key lv_node_key.
    ENDLOOP.
  ENDLOOP.

  " 트리 업데이트
  gcl_tree->expand_node( i_node_key = gv_all_node ).
  gcl_tree->update_calculations( ).
  gcl_tree->frontend_update( ).

ENDFORM.

FORM filter_detail  USING pv_node_key TYPE lvc_nkey.

  DATA : ls_node  LIKE gs_data,
         ls_item  LIKE LINE OF gt_item,
         lr_anlkl TYPE RANGE OF zte3cm0002-anlkl,
         lv_anlkl TYPE zte3cm0002-anlkl.

  CLEAR : gr_anln1, gr_anlkl.

  CALL METHOD gcl_tree->get_outtab_line
    EXPORTING
      i_node_key    = pv_node_key
    IMPORTING
      e_outtab_line = ls_node.

  IF pv_node_key NE gv_all_node.

    IF ls_node-anln1 IS NOT INITIAL.
      gr_anln1 = VALUE #(
      ( sign = 'I' option = 'EQ' low = ls_node-anln1 )
      ).
    ENDIF.

    IF ls_node-anlkl IS NOT INITIAL.
      CASE ls_node-anlkl.
        WHEN '건물'.
          lv_anlkl = 'BD'.
        WHEN '기계장치'.
          lv_anlkl = 'MC'.
        WHEN '운반구'.
          lv_anlkl = 'VH'.
        WHEN '비품'.
          lv_anlkl = 'FE'.
        WHEN '토지'.
          lv_anlkl = 'LD'.
        WHEN OTHERS.
          lv_anlkl = ls_node-anlkl.
      ENDCASE.
      gr_anlkl = VALUE #( ( sign = 'I' option = 'EQ' low = lv_anlkl ) ).

    ENDIF.
  ENDIF.

  APPEND LINES OF gr_anlkl TO lr_anlkl.
  IF pa_anlkl IS NOT INITIAL AND pa_anlkl NE 'ALL'.
    APPEND VALUE #( sign = 'I' option = 'EQ' low = pa_anlkl ) TO lr_anlkl.
  ENDIF.

  REFRESH gt_item.

  IF lr_anlkl IS INITIAL.
*   전체/자산번호 클릭 → anlkl 조건 없음, anln1만
    SELECT anlkl, bukrs, anln1, anlnt, waers, abflg, kostl,
           aprice, accum, bookvalue, alife, abudat, afasl,
           m01, m02, m03, m04, m05, m06,
           m07, m08, m09, m10, m11, m12
      FROM zcds_e3_cm_0001( p_gjahr = @pa_gjahr )
     WHERE bukrs = @pa_bukrs
       AND anln1  IN @gr_anln1
       AND abflg  NE 'X'
      INTO CORRESPONDING FIELDS OF TABLE @gt_item.
  ELSE.
*   자산분류 클릭 → anlkl + anln1 둘 다
    SELECT anlkl, bukrs, anln1, anlnt, waers, abflg, kostl,
           aprice, accum, bookvalue, alife, abudat, afasl,
           m01, m02, m03, m04, m05, m06,
           m07, m08, m09, m10, m11, m12
      FROM zcds_e3_cm_0001( p_gjahr = @pa_gjahr )
     WHERE bukrs = @pa_bukrs
       AND anln1  IN @gr_anln1
       AND anlkl  IN @lr_anlkl
       AND abflg  NE 'X'
      INTO CORRESPONDING FIELDS OF TABLE @gt_item.
  ENDIF.

  SORT gt_item BY anln1.
  PERFORM set_chart.
  gcl_grid->refresh_table_display(
    EXPORTING
      is_stable      = gs_stable
      i_soft_refresh = space ).

ENDFORM.

                        CHANGING pv_buzei pt_item.

  DATA : lv_loss TYPE p DECIMALS 2.

  IF pv_gain_loss < 0.

  ENDIF.

  ENDCASE.

    FROM zte3cm0002

   WHERE anln1 LIKE @lv_pattern.
