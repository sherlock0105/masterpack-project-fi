FORM create_object .

  DATA: lcl_bottom_con  TYPE REF TO cl_gui_container,
        lcl_lr_splitter TYPE REF TO cl_gui_splitter_container.

  CREATE OBJECT gcl_container
    EXPORTING
      repid     = sy-repid
      dynnr     = sy-dynnr
      side      = gcl_container->dock_at_left
      extension = 5000.

  " 전체 상하 분할 (TOP 20% / 하단 80%)
  CREATE OBJECT gcl_splitter_main
    EXPORTING
      parent  = gcl_container
      rows    = 2
      columns = 1
    EXCEPTIONS
      OTHERS  = 1.

  gcl_splitter_main->get_container(
    EXPORTING row = 1 column = 1
    RECEIVING container = gcl_container_top ).
  gcl_splitter_main->get_container(
    EXPORTING row = 2 column = 1
    RECEIVING container = lcl_bottom_con ).
  gcl_splitter_main->set_row_height( id = 1 height = 15 ).
  gcl_splitter_main->set_row_height( id = 2 height = 85 ).

  " 하단 좌우 분할 (트리 40% / 우측 60%)
  CREATE OBJECT lcl_lr_splitter
    EXPORTING
      parent  = lcl_bottom_con
      rows    = 1
      columns = 2
    EXCEPTIONS
      OTHERS  = 1.

  lcl_lr_splitter->get_container(
    EXPORTING row = 1 column = 1
    RECEIVING container = gcl_container_tree ).

  " 우측 임시 컨테이너
  DATA: lcl_right_con TYPE REF TO cl_gui_container.
  lcl_lr_splitter->get_container(
    EXPORTING row = 1 column = 2
    RECEIVING container = lcl_right_con ).

  lcl_lr_splitter->set_column_width( id = 1 width = 41 ).

  " 우측 상하 분할 (그리드 55% / 차트 45%)
  CREATE OBJECT gcl_splitter_right
    EXPORTING
      parent  = lcl_right_con
      rows    = 2
      columns = 1
    EXCEPTIONS
      OTHERS  = 1.

  gcl_splitter_right->get_container(
    EXPORTING row = 1 column = 1
    RECEIVING container = gcl_container_grid ).
  gcl_splitter_right->get_container(
    EXPORTING row = 2 column = 1
    RECEIVING container = gcl_container_chart ).
  gcl_splitter_right->set_row_height( id = 1 height = 55 ).
*  gcl_splitter_right->set_row_height( id = 2 height = 45 ).

  CREATE OBJECT gcl_tree
    EXPORTING
      parent              = gcl_container_tree
      node_selection_mode = cl_gui_column_tree=>node_sel_mode_multiple
      item_selection      = space
      no_html_header      = 'X'
      no_toolbar          = space.

  CALL METHOD gcl_tree->set_table_for_first_display
    EXPORTING
      i_save              = 'A'
      i_default           = 'X'
      is_variant          = gs_variant
      is_hierarchy_header = gs_hierhdr
    CHANGING
      it_outtab           = gt_data
      it_fieldcatalog     = gt_fcat.

  " 우측 상단: ALV 그리드
  CREATE OBJECT gcl_grid
    EXPORTING
      i_parent = gcl_container_grid.

  " 차트 오브젝트 생성
  CREATE OBJECT gcl_chart
    EXPORTING
      parent = gcl_container_chart.

  " 이벤트 등록
  CALL METHOD gcl_tree->get_registered_events
    IMPORTING
      events = gt_event.

  gs_event-eventid    = cl_gui_column_tree=>eventid_node_double_click.
  gs_event-appl_event = 'X'.
  APPEND gs_event TO gt_event.

    EXPORTING

      i_default       = 'X'

FORM get_data .

  DATA: lv_curr_to TYPE dats,
        lv_prev_to TYPE zcds_e3_fi_0011-gjahr.

  DATA: BEGIN OF gs_master,
          saknr       TYPE saknr,
          stext       TYPE txt20,
          major_class TYPE zte3fi0001-major_class,
          minor_class TYPE zte3fi0001-minor_class,
        END OF gs_master.
  DATA: lt_master LIKE TABLE OF gs_master.

  DATA: BEGIN OF gs_cds,
          saknr    TYPE saknr,
          curr_amt TYPE zte3fi0003-dmbtr,
          prev_amt TYPE zte3fi0003-dmbtr,
        END OF gs_cds.
  DATA: lt_cds LIKE TABLE OF gs_cds.

  REFRESH gt_data.

*  lv_curr_to = |{ pa_gjahr }1231|.
  lv_prev_to = |{ pa_gjahr - 1 }|.

  " 계정 마스터 조회
  SELECT saknr, stext, major_class, minor_class
    FROM zte3fi0001
   WHERE bukrs EQ @pa_bukrs
    INTO CORRESPONDING FIELDS OF TABLE @lt_master.

  " CDS 조회 (당기/전기 잔액)
  SELECT *
    FROM zcds_e3_fi_0011(
           p_bukrs    = @pa_bukrs,
           p_gjahr  = @pa_gjahr,
           p_prev_gjahr = @lv_prev_to
         )
    INTO CORRESPONDING FIELDS OF TABLE @lt_cds.

  SORT lt_cds BY saknr.

  " 마스터 기준 LOOP
  LOOP AT lt_master INTO gs_master.
    CLEAR gs_data.

    gs_data-saknr      = gs_master-saknr.
    gs_data-stext      = gs_master-stext.
    gs_data-waers      = 'KRW'.
    gs_data-waers_prev = 'KRW'.

    READ TABLE lt_cds INTO gs_cds
      WITH KEY saknr = gs_master-saknr BINARY SEARCH.
    IF sy-subrc EQ 0.
      gs_data-curr_amt = gs_cds-curr_amt / 1000.
      gs_data-prev_amt = gs_cds-prev_amt / 1000.
    ENDIF.

    CASE gs_master-minor_class.
      WHEN '11'.
        gs_data-fs_lv1  = '자산'.
        gs_data-fs_lv2  = '유동자산'.
        gs_data-fs_sort = '010'.
      WHEN '13'.
        gs_data-fs_lv1  = '자산'.
        gs_data-fs_lv2  = '비유동자산'.
        gs_data-fs_sort = '020'.
      WHEN '21'.
        gs_data-fs_lv1  = '부채'.
        gs_data-fs_lv2  = '유동부채'.
        gs_data-fs_sort = '030'.
      WHEN '23'.
        gs_data-fs_lv1  = '부채'.
        gs_data-fs_lv2  = '비유동부채'.
        gs_data-fs_sort = '040'.
      WHEN '31'.
        gs_data-fs_lv1  = '자본'.
        gs_data-fs_lv2  = '납입자본'.
        gs_data-fs_sort = '050'.
      WHEN '32'.
        gs_data-fs_lv1  = '자본'.
        gs_data-fs_lv2  = '이익잉여금'.
        gs_data-fs_sort = '051'.
      WHEN '33'.
        gs_data-fs_lv1  = '자본'.
        gs_data-fs_lv2  = '기타포괄손익누계액'.
        gs_data-fs_sort = '052'.
      WHEN OTHERS.
        CONTINUE.
    ENDCASE.

    APPEND gs_data TO gt_data.
  ENDLOOP.

  SORT gt_data BY fs_sort saknr.


ENDFORM.

*&---------------------------------------------------------------------*
FORM make_hierarchy .

  DATA: ls_empty        LIKE gs_data,
        ls_folder       LIKE gs_data,
        lt_data         LIKE TABLE OF gs_data,
        lv_root_key     TYPE lvc_nkey,   " 전체 루트
        lv_asset        TYPE lvc_nkey,
        lv_liab         TYPE lvc_nkey,
        lv_equity       TYPE lvc_nkey,
        lv_level_1      TYPE lvc_nkey,
        lv_level_2      TYPE lvc_nkey,
        lv_node_key     TYPE lvc_nkey,
        lv_cur_asset    TYPE lvc_nkey,
        lv_noncur_asset TYPE lvc_nkey,
        lv_cur_liab     TYPE lvc_nkey,
        lv_noncur_liab  TYPE lvc_nkey,
        lv_eq1          TYPE lvc_nkey,
        lv_eq2          TYPE lvc_nkey,
        lv_eq3          TYPE lvc_nkey,
        lv_leaf         TYPE char80.

  APPEND LINES OF gt_data TO lt_data.
*  REFRESH gt_data.

  " 전체 루트 폴더
  CLEAR ls_folder.
  PERFORM add_node USING ls_folder '전체' '' lv_root_key abap_true.

  " 자산
  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자산'.
  PERFORM add_node USING ls_folder '자산' lv_root_key lv_asset abap_true.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자산'.
  ls_folder-fs_lv2 = '유동자산'.
  PERFORM add_node USING ls_folder '유동자산' lv_asset lv_level_1 abap_true.
  lv_cur_asset = lv_level_1.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자산'.
  ls_folder-fs_lv2 = '비유동자산'.
  PERFORM add_node USING ls_folder '비유동자산' lv_asset lv_level_1 abap_true.
  lv_noncur_asset = lv_level_1.

  " 부채
  CLEAR ls_folder.
  ls_folder-fs_lv1 = '부채'.
  PERFORM add_node USING ls_folder '부채' lv_root_key lv_liab abap_true.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '부채'.
  ls_folder-fs_lv2 = '유동부채'.
  PERFORM add_node USING ls_folder '유동부채' lv_liab lv_level_1 abap_true.
  lv_cur_liab = lv_level_1.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '부채'.
  ls_folder-fs_lv2 = '비유동부채'.
  PERFORM add_node USING ls_folder '비유동부채' lv_liab lv_level_1 abap_true.
  lv_noncur_liab = lv_level_1.

  " 자본
  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자본'.
  PERFORM add_node USING ls_folder '자본' lv_root_key lv_equity abap_true.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자본'.
  ls_folder-fs_lv2 = '납입자본'.
  PERFORM add_node USING ls_folder '납입자본' lv_equity lv_level_1 abap_true.
  lv_eq1 = lv_level_1.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자본'.
  ls_folder-fs_lv2 = '이익잉여금'.
  PERFORM add_node USING ls_folder '이익잉여금' lv_equity lv_level_1 abap_true.
  lv_eq2 = lv_level_1.

  CLEAR ls_folder.
  ls_folder-fs_lv1 = '자본'.
  ls_folder-fs_lv2 = '기타포괄손익누계액'.
  PERFORM add_node USING ls_folder '기타포괄손익누계액' lv_equity lv_level_1 abap_true.
  lv_eq3 = lv_level_1.

  " 계정 리프 노드
  LOOP AT lt_data INTO gs_data.
    lv_leaf = gs_data-stext.
    CASE gs_data-fs_lv2.
      WHEN '유동자산'.
        lv_level_2 = lv_cur_asset.
      WHEN '비유동자산'.
        lv_level_2 = lv_noncur_asset.
      WHEN '유동부채'.
        lv_level_2 = lv_cur_liab.
      WHEN '비유동부채'.
        lv_level_2 = lv_noncur_liab.
      WHEN '납입자본'.
        lv_level_2 = lv_eq1.
      WHEN '이익잉여금'.
        lv_level_2 = lv_eq2.
      WHEN '기타포괄손익누계액'.
        lv_level_2 = lv_eq3.
      WHEN OTHERS. CONTINUE.
    ENDCASE.
    PERFORM add_node USING gs_data lv_leaf lv_level_2 lv_node_key abap_false.
  ENDLOOP.

  " 루트만 펼치고 하위는 접힌 상태
  CALL METHOD gcl_tree->expand_node
    EXPORTING
      i_node_key = lv_root_key.

  gcl_tree->expand_node(
  EXPORTING
    i_node_key          =  lv_asset ).

  gcl_tree->expand_node(
    EXPORTING
      i_node_key          =  lv_equity ).                " Node Key

  gcl_tree->expand_node(
    EXPORTING
      i_node_key          =  lv_liab ).

  CALL METHOD gcl_tree->update_calculations.
  CALL METHOD gcl_tree->frontend_update.

  " 초기 전체 데이터 그리드/차트 표시
  PERFORM load_all_data.

ENDFORM.

*& Form filter_detail
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> NODE_KEY
*&---------------------------------------------------------------------*
FORM filter_detail  USING pv_node_key TYPE lvc_nkey.

  DATA: ls_node_data  LIKE gs_data,
        lr_saknr      TYPE RANGE OF zte3fi0001-saknr,
        ls_saknr      LIKE LINE OF lr_saknr,
        lv_prev_gjahr TYPE gjahr.

  CALL METHOD gcl_tree->get_outtab_line
    EXPORTING
      i_node_key    = pv_node_key
    IMPORTING
      e_outtab_line = ls_node_data.

  REFRESH: gt_header, gt_prev, lr_saknr.

  " saknr RANGE 구성
  IF ls_node_data-saknr IS NOT INITIAL.
    ls_saknr-sign   = 'I'.
    ls_saknr-option = 'EQ'.
    ls_saknr-low    = ls_node_data-saknr.
    APPEND ls_saknr TO lr_saknr.
  ELSEIF ls_node_data-fs_lv2 IS NOT INITIAL.
    LOOP AT gt_data INTO gs_data
      WHERE fs_lv2 EQ ls_node_data-fs_lv2.
      ls_saknr-sign   = 'I'.
      ls_saknr-option = 'EQ'.
      ls_saknr-low    = gs_data-saknr.
      APPEND ls_saknr TO lr_saknr.
    ENDLOOP.
  ELSEIF ls_node_data-fs_lv1 IS NOT INITIAL.
    LOOP AT gt_data INTO gs_data
      WHERE fs_lv1 EQ ls_node_data-fs_lv1.
      ls_saknr-sign   = 'I'.
      ls_saknr-option = 'EQ'.
      ls_saknr-low    = gs_data-saknr.
      APPEND ls_saknr TO lr_saknr.
    ENDLOOP.
  ELSE.
    LOOP AT gt_data INTO gs_data.
      ls_saknr-sign   = 'I'.
      ls_saknr-option = 'EQ'.
      ls_saknr-low    = gs_data-saknr.
      APPEND ls_saknr TO lr_saknr.
    ENDLOOP.
  ENDIF.

  CHECK lr_saknr IS NOT INITIAL.

  " 당기 조회
  SELECT saknr, stext, waers,
         m01, m02, m03, m04, m05, m06,
         m07, m08, m09, m10, m11, m12
    FROM zcds_e3_fi_0016(
           p_bukrs = @pa_bukrs,
           p_gjahr = @pa_gjahr
         )
    INTO CORRESPONDING FIELDS OF TABLE @gt_header
   WHERE saknr IN @lr_saknr
   ORDER BY saknr.

  IF gt_header IS INITIAL.
    MESSAGE s000 WITH '올해 데이터가 없습니다.'.
    RETURN.
  ENDIF.


  " 전기 조회
  lv_prev_gjahr = pa_gjahr - 1.

  SELECT saknr, stext, waers,
         m01, m02, m03, m04, m05, m06,
         m07, m08, m09, m10, m11, m12
    FROM zcds_e3_fi_0016(
           p_bukrs = @pa_bukrs,
           p_gjahr = @lv_prev_gjahr
         )
    INTO CORRESPONDING FIELDS OF TABLE @gt_prev
   WHERE saknr IN @lr_saknr
   ORDER BY saknr.


  CALL METHOD gcl_grid->refresh_table_display
    EXPORTING
      i_soft_refresh = 'X'.

  PERFORM set_chart.

ENDFORM.

*& Form set_chart
*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM set_chart .

  CHECK gcl_chart IS BOUND.

  DATA: lv_xdata   TYPE xstring,
        lv_xcustom TYPE xstring,
        lv_val     TYPE string.

  DATA: lv_root     TYPE REF TO if_ixml_element,
        lv_cats     TYPE REF TO if_ixml_element,
        lv_cat      TYPE REF TO if_ixml_element,
        lv_ser_prev TYPE REF TO if_ixml_element,
        lv_ser_curr TYPE REF TO if_ixml_element,
        lv_point    TYPE REF TO if_ixml_element,
        lv_value    TYPE REF TO if_ixml_element.

  DATA: lo_ixml   TYPE REF TO if_ixml,
        lo_sf     TYPE REF TO if_ixml_stream_factory,
        lo_enc    TYPE REF TO if_ixml_encoding,
        lo_doc_c  TYPE REF TO if_ixml_document,
        lo_root_c TYPE REF TO if_ixml_element,
        lo_global TYPE REF TO if_ixml_element.

  " 자산/부채/자본 전기/당기 합계
  DATA: lv_asset_prev  TYPE p DECIMALS 2,
        lv_asset_curr  TYPE p DECIMALS 2,
        lv_liab_prev   TYPE p DECIMALS 2,
        lv_liab_curr   TYPE p DECIMALS 2,
        lv_equity_prev TYPE p DECIMALS 2,
        lv_equity_curr TYPE p DECIMALS 2.

  " gt_data에서 집계 (트리 데이터 그대로 활용)
  LOOP AT gt_data INTO gs_data.
    CASE gs_data-fs_lv1.
      WHEN '자산'.
        lv_asset_prev  = lv_asset_prev  + abs( gs_data-prev_amt ).
        lv_asset_curr  = lv_asset_curr  + abs( gs_data-curr_amt ).
      WHEN '부채'.
        lv_liab_prev   = lv_liab_prev   + abs( gs_data-prev_amt ).
        lv_liab_curr   = lv_liab_curr   + abs( gs_data-curr_amt ).
      WHEN '자본'.
        lv_equity_prev = lv_equity_prev + abs( gs_data-prev_amt ).
        lv_equity_curr = lv_equity_curr + abs( gs_data-curr_amt ).
    ENDCASE.
  ENDLOOP.

  " ① Customizing XML (그룹 막대 - 전기/당기 나란히)
  lo_ixml  = cl_ixml=>create( ).
  lo_sf    = lo_ixml->create_stream_factory( ).
  lo_enc   = lo_ixml->create_encoding(
               byte_order    = if_ixml_encoding=>co_little_endian
               character_set = 'utf-8' ).
  lo_doc_c = lo_ixml->create_document( ).
  lo_doc_c->set_encoding( lo_enc ).

  lo_root_c = lo_doc_c->create_simple_element(
                name = 'SAPChartCustomizing' parent = lo_doc_c ).
  lo_root_c->set_attribute( name = 'version' value = '3.0' ).
  lo_global = lo_doc_c->create_simple_element(
                name = 'GlobalSettings' parent = lo_root_c ).
  lo_doc_c->create_simple_element(
    name   = 'ChartType'
    parent = lo_global )->if_ixml_node~set_value( 'Columns' ).  " 그룹 세로막대

  lo_doc_c->render( ostream = lo_sf->create_ostream_xstring( lv_xcustom ) ).
  gcl_chart->set_customizing( xdata = lv_xcustom ).

  " ② Data XML
  gcl_ixml          = cl_ixml=>create( ).
  gcl_ixml_sf       = gcl_ixml->create_stream_factory( ).
  gcl_ixml_data_doc = gcl_ixml->create_document( ).
  gcl_encoding      = gcl_ixml->create_encoding(
                        byte_order    = if_ixml_encoding=>co_little_endian
                        character_set = 'utf-8' ).
  gcl_ixml_data_doc->set_encoding( gcl_encoding ).

  lv_root = gcl_ixml_data_doc->create_simple_element(
              name = 'ChartData' parent = gcl_ixml_data_doc ).
  lv_cats = gcl_ixml_data_doc->create_simple_element(
              name = 'Categories' parent = lv_root ).

  " 시리즈: 전기 / 당기
  lv_ser_prev = gcl_ixml_data_doc->create_simple_element(
                  name = 'Series' parent = lv_root ).
  lv_ser_prev->set_attribute( name = 'label'
                               value = |전기({ pa_gjahr - 1 })| ).

  lv_ser_curr = gcl_ixml_data_doc->create_simple_element(
                  name = 'Series' parent = lv_root ).
  lv_ser_curr->set_attribute( name = 'label'
                               value = |당기({ pa_gjahr })| ).

  " ③ 카테고리/포인트 매크로
  DEFINE _add_col.
    lv_cat = gcl_ixml_data_doc->create_simple_element(
               name = 'Category' parent = lv_cats ).
    lv_cat->if_ixml_node~set_value( &1 ).
    " 전기
    lv_val = |{ &2 * 100 NUMBER = RAW }|.
    lv_point = gcl_ixml_data_doc->create_simple_element(
                 name = 'Point' parent = lv_ser_prev ).
    lv_value = gcl_ixml_data_doc->create_simple_element(
                 name = 'Value' parent = lv_point ).
    lv_value->set_attribute( name = 'type' value = 'y' ).
    lv_value->if_ixml_node~set_value( lv_val ).
    " 당기
    lv_val = |{ &3 * 100 NUMBER = RAW }|.
    lv_point = gcl_ixml_data_doc->create_simple_element(
                 name = 'Point' parent = lv_ser_curr ).
    lv_value = gcl_ixml_data_doc->create_simple_element(
                 name = 'Value' parent = lv_point ).
    lv_value->set_attribute( name = 'type' value = 'y' ).
    lv_value->if_ixml_node~set_value( lv_val ).
  END-OF-DEFINITION.

  _add_col '자산' lv_asset_prev  lv_asset_curr.
  _add_col '부채' lv_liab_prev   lv_liab_curr.
  _add_col '자본' lv_equity_prev lv_equity_curr.

  gcl_ixml_data_doc->render(
    ostream = gcl_ixml_sf->create_ostream_xstring( lv_xdata ) ).

  gcl_chart->set_data( xdata = lv_xdata ).
  gcl_chart->render( ).

ENDFORM.
