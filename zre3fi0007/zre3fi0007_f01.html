FORM start_data . "감가상각 실행

  DATA: lv_calc_depr TYPE p DECIMALS 4,
        lv_gjahr     TYPE zte3fi0002-gjahr,
        lv_monat     TYPE zte3fi0002-monat.

  lv_gjahr = sy-datum(4).
  lv_monat = sy-datum+4(2).

  REFRESH: gt_data, gt_depr.

  SELECT *
    FROM zte3cm0002
    INTO CORRESPONDING FIELDS OF TABLE @gt_data
   WHERE bukrs = @gv_bukrs
     AND afasl NE 'NONE'.

  IF gt_data IS INITIAL.
    MESSAGE s000 WITH '상각 대상 데이터가 존재하지 않습니다.'.
    RETURN.
  ENDIF.

*  당월 중복 실행 체크
  SELECT COUNT(*)
    FROM zte3cm0007
   WHERE bukrs = @gv_bukrs
     AND gjahr = @lv_gjahr
     AND monat = @lv_monat.

  IF sy-dbcnt > 0.
    MESSAGE s000 WITH lv_gjahr '년' lv_monat '월 감가상각이 이미 처리되었습니다.'.
    RETURN.
  ENDIF.

  " 감가상각 계산 루프
  LOOP AT gt_data ASSIGNING FIELD-SYMBOL(<ls_data>).

    CLEAR: lv_calc_depr, gs_depr.

    " 상각방법별 계산
    CASE <ls_data>-afasl.
      WHEN 'LINR'. " 정액법: 취득원가 / 내용연수 / 12
        IF <ls_data>-alife > 0.
          lv_calc_depr = <ls_data>-aprice / <ls_data>-alife / 12.
        ENDIF.

      WHEN 'DECL'. " 정률법: (취득원가 - 누계액) * (2 / 내용연수) / 12
        IF <ls_data>-alife > 0.
          lv_calc_depr = ( <ls_data>-aprice - <ls_data>-accum ) * ( 2 / <ls_data>-alife ) / 12.
        ENDIF.
    ENDCASE.

    " 잔액 초과 방지
    IF <ls_data>-accum + lv_calc_depr > <ls_data>-aprice.
      lv_calc_depr = <ls_data>-aprice - <ls_data>-accum.
    ENDIF.

    " 이력 테이블
    gs_depr-mandt = sy-mandt.
    gs_depr-bukrs = <ls_data>-bukrs.
    gs_depr-gjahr = lv_gjahr.
    gs_depr-monat = lv_monat.
    gs_depr-anln1 = <ls_data>-anln1.
    gs_depr-depr  = lv_calc_depr.                     " 당월 상각액
    gs_depr-accum = <ls_data>-accum + lv_calc_depr.   " 새로운 누계액
    gs_depr-waers = <ls_data>-waers.
    gs_depr-erdat = sy-datum.
    gs_depr-erzet = sy-uzeit.
    gs_depr-ernam = sy-uname.

    APPEND gs_depr TO gt_depr.

    " 마스터 테이블 갱신
    <ls_data>-accum = gs_depr-accum.

  ENDLOOP.

  IF gt_depr IS NOT INITIAL.

    MODIFY zte3cm0007 FROM TABLE gt_depr. " 이력 적재
    MODIFY zte3cm0002 FROM TABLE gt_data. " 마스터 갱신

    IF sy-subrc EQ 0.

      "전표 생성
      PERFORM post_document USING lv_gjahr lv_monat.

      IF gv_belnr IS NOT INITIAL.
        MESSAGE s000 WITH | { lv_monat } 월 감가상각 처리 완료 |.
      ENDIF.

    ELSE.
      MESSAGE s000 WITH '데이터 저장 실패' DISPLAY LIKE 'E'.
      ROLLBACK WORK.

    ENDIF.
  ENDIF.

ENDFORM.

FORM check_uname .

  gv_bukrs = '1000'.
  gv_pernr = '10000207'.

  IF sy-uname NE 'NCODE-E-05'.
    LEAVE PROGRAM.
  ENDIF.

ENDFORM.

FORM post_document  USING    pv_gjahr TYPE zte3fi0002-gjahr
                             pv_monat TYPE zte3fi0002-monat.


  " 변수 선언 - LOOP 내 DATA() 인라인 금지
  DATA: lt_item   TYPE TABLE OF zte3fi0003,
        ls_item   TYPE zte3fi0003,
        ls_depr   TYPE zte3cm0007,   " ← LOOP INTO 대상 선언
        ls_master TYPE zte3cm0002,   " ← READ TABLE INTO 대상 선언
        ls_header TYPE zte3fi0002,
        lv_date   TYPE sy-datum.

  CALL FUNCTION 'RP_LAST_DAY_OF_MONTHS'
    EXPORTING
      day_in            = sy-datum                " Key date
    IMPORTING
      last_day_of_month = lv_date.                 " Date of last day of the month from key  date


  ls_header = VALUE zte3fi0002(
                  bukrs = gv_bukrs
                  blart = 'SA'
                  waers = 'KRW'
                  budat = lv_date   "월말일을 계산
                  bldat = lv_date
                  pernr = gv_pernr
                  bltxt = |{ pv_gjahr }년 { pv_monat }월 감가상각|
  ).

  LOOP AT gt_depr INTO ls_depr WHERE depr > 0.

    READ TABLE gt_data INTO ls_master WITH KEY anln1 = ls_depr-anln1.
    "차변
    ls_item = VALUE #(
      saknr = '5200000009' "감가상각비
      shkzg = 'S'
      dmbtr = ls_depr-depr
      kostl = ls_master-kostl
    ).
    APPEND ls_item TO lt_item.
    "대변
    ls_item = VALUE #(
      saknr = '1300000006' "감가상각누계액
      shkzg = 'H'
      dmbtr = ls_depr-depr
      anln1 = ls_depr-anln1
    ).
    APPEND ls_item TO lt_item.

  ENDLOOP.

  CHECK lt_item IS NOT INITIAL.

  zcl_e3_fi_document=>post_document(
      EXPORTING is_header = ls_header
                it_items  = lt_item
      IMPORTING
                ev_belnr  = gv_belnr
                ).

ENDFORM.
