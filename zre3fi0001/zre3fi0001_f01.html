FORM get_api .

  gv_date = sy-datum.

  DO 5 TIMES.
    CLEAR: gv_json_data, gt_json.

    gv_url = |https://oapi.koreaexim.go.kr/site/program/financial/exchangeJSON?| &&
             |authkey={ gc_key }&searchdate={ gv_date }&data=AP01|.

    CALL METHOD cl_http_client=>create_by_url
      EXPORTING
        url    = gv_url
      IMPORTING
        client = go_http_client.

    go_http_client->request->set_method( if_http_request=>co_request_method_get ).
    go_http_client->send( ).
    go_http_client->receive( ).
    gv_json_data = go_http_client->response->get_cdata( ).
    go_http_client->close( ).


    /ui2/cl_json=>deserialize(
      EXPORTING json = gv_json_data
      CHANGING  data = gt_json ).

    " 결과 확인 (성공 코드: 1)
    READ TABLE gt_json INTO gs_json INDEX 1.
    IF sy-subrc = 0 AND gs_json-result = 1.
      EXIT.
    ELSEIF sy-subrc = 0 AND gs_json-result = 3.
      MESSAGE 'API 인증 오류: 키를 확인하세요.' TYPE 'E'.
      STOP.
    ENDIF.

    gv_date = gv_date - 1. " 하루 전으로 재시도
  ENDDO.

ENDFORM.

FORM save_data.
  REFRESH gt_save.

*  IF gv_json_data IS INITIAL OR gv_json_data = '[]'.
*    MESSAGE '수신된 환율 데이터가 없습니다. (최근 5일 영업일 확인 필요)' TYPE 'S' DISPLAY LIKE 'W'.
*    STOP.
*  ENDIF.

  " 인버전 데이트 계산 (SAP 환율 테이블 표준 방식)
  gv_gdatu_inv = 99999999 - sy-datum.

  LOOP AT gt_json INTO gs_json.
    " 1. 통화 필터링 (본인 요청: USD, EUR, JPY)
    CHECK gs_json-cur_unit = 'USD' OR
          gs_json-cur_unit = 'EUR' OR
          gs_json-cur_unit CS 'JPY'.

    " 2. 콤마 제거 및 공백 정리
    REPLACE ALL OCCURRENCES OF ',' IN gs_json-deal_bas_r WITH ''.
    CONDENSE gs_json-deal_bas_r NO-GAPS.

    " 3. 데이터 매핑 시작
    CLEAR gs_save.
    gs_save-mandt = sy-mandt.
    gs_save-kurst = 'M'.
    gs_save-tcurr = 'KRW'.
    gs_save-gdatu = gv_gdatu_inv.

    " 4. 통화 단위 및 Factor 처리 (가장 중요)
    IF gs_json-cur_unit CS '(100)'.
      gs_save-fcurr = gs_json-cur_unit+0(3). " JPY 등
      gs_save-ffact = 100.
    ELSE.
      gs_save-fcurr = gs_json-cur_unit.
      gs_save-ffact = 1.
    ENDIF.
    gs_save-tfact = 1.

    " 5. 환율값 대입 (trunc 제거하여 소수점 정확도 유지)
    IF gs_json-deal_bas_r CO '0123456789. '.
      gs_save-ukurs = trunc( gs_json-deal_bas_r ).
    ELSE.
      CONTINUE. " 숫자가 아니면 해당 건은 Skip (덤프 방지)
    ENDIF.

    " 타임스탬프
    gs_save-erdat = sy-datum.
    gs_save-erzet = sy-uzeit.
    gs_save-ernam = sy-uname.

    APPEND gs_save TO gt_save.
  ENDLOOP.

  " 6. DB 반영
  IF gt_save IS NOT INITIAL.
    MODIFY zte3fi0008 FROM TABLE gt_save. " 본인 테이블명 유지

    IF sy-subrc = 0.
      COMMIT WORK.
      MESSAGE |환율 { lines( gt_save ) }건 저장 완료 (기준일: { gv_date })| TYPE 'S'.
    ELSE.
      ROLLBACK WORK.
      MESSAGE 'DB 저장 중 오류가 발생했습니다.' TYPE 'E'.
    ENDIF.
  ENDIF.
ENDFORM.
