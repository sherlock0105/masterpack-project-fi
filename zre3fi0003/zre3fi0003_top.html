DEFINE check_date_range.

*  IF &1-low IS INITIAL OR &1-high IS INITIAL.
*    MESSAGE s000 WITH '조회기간을 입력하세요' DISPLAY LIKE 'E'.
*    STOP.
*  ENDIF.

  IF &1-low IS NOT INITIAL OR &1-high IS NOT INITIAL.
    CLEAR lv_limit.
    CALL FUNCTION 'RP_CALC_DATE_IN_INTERVAL'
      EXPORTING
        date      = &1-low
        days      = 0
        months    = 3
        signum    = '+'
        years     = 0
      IMPORTING
        calc_date = &2.

    IF &1-high > lv_limit.
      MESSAGE s000 WITH '조회 기간은 최대 3개월까지만 가능합니다.' DISPLAY LIKE 'E'.
      STOP.
    ENDIF.
  ENDIF.
