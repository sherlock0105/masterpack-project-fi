FORM get_header_data .

  REFRESH gt_header.

  CASE gv_tab.

    WHEN 'COMM1'.  " 전체 전표
      gv_mode = 'ALL'.

    WHEN 'COMM2'.  " 고객 전표
      gv_mode = COND #( WHEN rb_all = 'X' THEN 'AR_A'
                        WHEN rb_opn = 'X' THEN 'AR_O'
                        WHEN rb_clr = 'X' THEN 'AR_C' ).

    WHEN 'COMM3'.  " 공급업체
      gv_mode = COND #( WHEN rb_all2 = 'X' THEN 'AP_A'
                        WHEN rb_opn2 = 'X' THEN 'AP_O'
                        WHEN rb_clr2 = 'X' THEN 'AP_C' ).

    WHEN 'COMM4'.  " GL 전표
      gv_mode = 'GL'.

    WHEN OTHERS.   "탭이 기본일때
      gv_mode = 'ALL'.
      gv_tab = 'COMM1'.

  ENDCASE.

  gv_top_flag = gv_tab. "상단 정보 표시용 플래그 저장
  "데이터 조회 및 후행 작업
  PERFORM get_data  USING gv_mode.
  PERFORM set_icon.
  PERFORM set_sort.

ENDFORM.

FORM set_icon .

  LOOP AT gt_header ASSIGNING FIELD-SYMBOL(<fs_header>).
    " 역분개
    IF <fs_header>-stblg IS NOT INITIAL OR <fs_header>-blart EQ 'AB'.
      <fs_header>-icon = icon_led_red.      " 역분개/취소됨
      <fs_header>-icon_text = '역분개'.
      CONTINUE. " 역분개된 전표는 아래 로직을 타지 않고 다음 loop로 이동
    ENDIF.
    " SA 전표 (일반 전표) 처리
    IF <fs_header>-blart EQ 'SA'.
      <fs_header>-icon = icon_led_inactive.
      <fs_header>-icon_text = '일반'.

      CONTINUE.
    ENDIF.
    " 그 외 전표 상태 설정
    IF <fs_header>-bstat EQ 'C'.
      <fs_header>-icon = icon_led_green.    " 반제(완료)
      <fs_header>-icon_text = '반제'.
    ELSEIF <fs_header>-open_flag EQ 'X'.
      <fs_header>-icon = icon_led_yellow.   " 미결
      <fs_header>-icon_text = '미결'.
    ELSE.
      <fs_header>-icon = icon_led_green.    " 일반
      <fs_header>-icon_text = '정상'.
    ENDIF.

  ENDLOOP.

ENDFORM.

*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
*&      --> E_ROW_ID
*&      --> E_COLUMN_ID
*&---------------------------------------------------------------------*
FORM handle_hotspot  USING   ps_row TYPE lvc_s_row
                             ps_col TYPE lvc_s_col.

  CHECK ps_row-rowtype IS INITIAL.

  READ TABLE gt_header INTO gs_header INDEX ps_row-index.
  CHECK sy-subrc EQ 0.

  CASE ps_col-fieldname.

    WHEN 'XBLNR'.
      PERFORM xblnr_popup USING gs_header-xblnr gs_header-gjahr.

    WHEN 'BELNR'.
      PERFORM get_data_item USING gs_header-belnr.

    WHEN 'KUNNR'.

      call function 'ZFE3FI0004'
        EXPORTING
          iv_kunnr = gs_header-kunnr.

    WHEN 'LIFNR'.

      IF gs_header-lifnr IS INITIAL.
        RETURN.
      ENDIF.

      call function 'ZFE3FI0005'
        EXPORTING
          iv_lifnr = gs_header-lifnr.          " [MM] 공급업체코드

    WHEN 'STBLG'.
      call function 'ZFE3FI0006'
        EXPORTING
          iv_belnr = gs_header-stblg           " [FI] 전표 번호
          iv_bukrs = '1000'                    " [FI] 회사코드
          iv_gjahr = gs_header-gjahr.          " [FI] 회계연도

    WHEN 'AUGBL'.
      READ TABLE gt_item_data INTO gs_item_data INDEX ps_row-index.
      CHECK sy-subrc EQ 0.
      call function 'ZFE3FI0006'
        EXPORTING
          iv_belnr = gs_item_data-augbl        " [FI] 전표 번호
          iv_bukrs = '1000'                    " [FI] 회사코드
          iv_gjahr = gs_header-gjahr.          " [FI] 회계연도

  ENDCASE.

  PERFORM refresh_grid_bot.

ENDFORM.

*&      --> GS_HEADER_BELNR

FORM get_data_item  USING  pv_belnr TYPE zte3fi0002-belnr.

  CLEAR : gv_dmbtr_h, gv_dmbtr_s.
  REFRESH gt_item_data.

  SELECT belnr buzei saknr kostl shkzg sgtxt
         wrbtr fwaer dmbtr waers augbl augdt
    INTO CORRESPONDING FIELDS OF TABLE gt_item_data
    FROM zte3fi0003
   WHERE belnr EQ pv_belnr
     AND gjahr EQ gs_header-gjahr
   ORDER BY belnr buzei shkzg.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH '아이템 데이터가 없습니다.'.
    RETURN.
  ENDIF.

  " 역분개 전표 아이템 추가 (stblg가 있을 경우)
  IF gs_header-stblg IS NOT INITIAL AND
     gs_header-blart NE 'AB'.
    " APPENDING TABLE을 사용하면 기존 테이블 뒤에 바로 붙음
    SELECT belnr buzei saknr kostl shkzg sgtxt
           wrbtr fwaer dmbtr waers augbl augdt
      APPENDING CORRESPONDING FIELDS OF TABLE gt_item_data
      FROM zte3fi0003
     WHERE belnr EQ gs_header-stblg
       AND gjahr EQ gs_header-stjah "추가
     ORDER BY belnr buzei.

  ENDIF.

  PERFORM sum_dmbtr_sh. "차/대 분리
  PERFORM set_sort.

ENDFORM.

FORM get_data  USING   pv_mode TYPE c.

  DATA : lr_blart TYPE RANGE OF zte3fi0002-blart.

  IF pa_blart IS NOT INITIAL.
    lr_blart = VALUE #( BASE lr_blart

                       ( low    = pa_blart
                         sign   = 'I'
                         option = 'EQ' )

                        ).
  ENDIF.

  CLEAR gt_header.

  CASE pv_mode.

    WHEN 'ALL'.  " 전체
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_bukrs
         AND gjahr EQ pa_gjahr
         AND budat IN so_budat
         AND bstat NE 'T'
         AND belnr NOT LIKE '9%'
         AND belnr IN so_belnr
         AND blart IN lr_blart.        " 전체 = 공백이면 조건 없음

    WHEN 'AR_A'.  " 고객 전체
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk2
         AND gjahr EQ pa_gja2
         AND kunnr IN so_kun
         AND kunnr NE space
         AND bstat NE 'T'
         AND lifnr EQ space
         AND budat IN so_bud2
         AND belnr NOT LIKE '9%'
         AND blart IN ('DR', 'DZ')
         AND blart NE 'AB'.            " 역분개 제외

    WHEN 'AR_O'.  " 고객 미결 — 승인(A)됐는데 반제(C) 아닌 것
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk2
         AND gjahr EQ pa_gja2
         AND kunnr IN so_kun
         AND budat IN so_bud2
         AND open_flag EQ 'X'
         AND lifnr EQ space
         AND bstat EQ ' '
         AND belnr NOT LIKE '9%'
         AND stblg EQ space
         AND blart NE 'AB'.

    WHEN 'AR_C'.  " 고객 반제
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk2
         AND gjahr EQ pa_gja2
         AND kunnr IN so_kun
         AND lifnr EQ space
         AND budat IN so_bud2
         AND belnr NOT LIKE '9%'
         AND stblg EQ space
         AND bstat EQ 'C'.             " 반제

    WHEN 'AP_A'.  " 공급업체 전체
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk3
         AND gjahr EQ pa_gja3
         AND lifnr IN so_lif
         AND lifnr NE space
         AND bstat NE 'T'
         AND belnr NOT LIKE '9%'
         AND kunnr EQ space
         AND budat IN so_bud3
        AND blart IN ('KR', 'RE', 'KZ', 'PR' )
         AND blart NE 'AB'.

    WHEN 'AP_O'.  " 공급업체 미결
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk3
         AND gjahr EQ pa_gja3
         AND lifnr IN so_lif
         AND lifnr NE space
         AND budat IN so_bud3
         AND bstat EQ ' '  "빈값
         AND bstat NE 'T'
         AND belnr NOT LIKE '9%'
         AND stblg EQ space
         AND open_flag EQ 'X'
         AND blart NE 'AB'.

    WHEN 'AP_C'.  " 공급업체 반제
      SELECT bukrs gjahr belnr blart budat bldat waers
             kunnr lifnr pernr stblg stjah stgrd
             xblnr bltxt bstat open_flag rev_date kursf
        INTO CORRESPONDING FIELDS OF TABLE gt_header
        FROM zte3fi0002
       WHERE bukrs EQ pa_buk3
         AND gjahr EQ pa_gja3

*&---------------------------------------------------------------------*
FORM save_rev .

  DATA: lv_belnr  TYPE zte3fi0002-belnr,
        lt_rev_db TYPE TABLE OF zte3fi0003,
        ls_rev_db TYPE zte3fi0003.
  " 채번
  CALL FUNCTION 'NUMBER_GET_NEXT'
    EXPORTING
      nr_range_nr = '02'
      object      = 'ZNRE3FI01'
    IMPORTING
      number      = lv_belnr.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH '전표 채번 실패.' DISPLAY LIKE 'E'. RETURN.
  ENDIF.

  SELECT buzei, bukrs, belnr, gjahr, saknr,
         koart, bschl, shkzg, kostl, wrbtr,
         fwaer, dmbtr, waers, ukurs, sgtxt,
         kunnr, lifnr, invno, pdnum, ebeln,
         matnr, augbl, augdt
    FROM zte3fi0003
    INTO CORRESPONDING FIELDS OF TABLE @gt_item
   WHERE bukrs = @gv_sel_bukrs
     AND gjahr = @gv_sel_gjahr
     AND belnr = @gv_sel_belnr.

  IF gt_item IS INITIAL.
    MESSAGE s000 WITH '원본 아이템 없음.' DISPLAY LIKE 'E'. RETURN.
  ENDIF.

  " 전표유형별 역분개 아이템 생성
  CASE gs_header-blart.
    WHEN 'DZ' OR 'KZ'.
      PERFORM set_clear.

  ENDCASE.

  PERFORM reverse_items USING lv_belnr.

  " 역분개 헤더 준비
  CLEAR gs_rev.
  gs_rev = CORRESPONDING #( gs_header ).
  gs_rev-belnr    = lv_belnr.
  gs_rev-bstat    = space.
  gs_rev-blart    = 'AB'.
  gs_rev-kursf    = gs_header-kursf.
  gs_rev-budat    = sy-datum.
  gs_rev-bldat    = sy-datum.
  gs_rev-stblg    = gv_sel_belnr.        " 원본 전표번호
  gs_rev-stjah    = gv_sel_gjahr.        " 원본 회계연도
  gs_rev-stgrd    = gv_stgrd.            " 역분개 사유코드
  gs_rev-rev_date = sy-datum.           " 역분개 처리일
  gs_rev-pernr    = gs_emp-pernr.        " 요청자 사번

  " 헤더 INSERT
  INSERT zte3fi0002 FROM @gs_rev.
  IF sy-subrc NE 0.
    MESSAGE s000 WITH '역분개 헤더 저장 실패.' DISPLAY LIKE 'E'.
    ROLLBACK WORK.
    RETURN.
  ENDIF.
  " gt_rev_item → zte3fi0003 타입 변환 후 INSERT
  LOOP AT gt_rev_item INTO DATA(ls_rev).
    CLEAR ls_rev_db.
    MOVE-CORRESPONDING ls_rev TO ls_rev_db.
    APPEND ls_rev_db TO lt_rev_db.

  ENDLOOP.

  INSERT zte3fi0003 FROM TABLE @lt_rev_db.
  IF sy-subrc NE 0.
    MESSAGE s000 WITH '역분개 아이템 저장 실패.' DISPLAY LIKE 'E'.
    ROLLBACK WORK.
    RETURN.
  ENDIF.

  " 원본 전표 STBLG + STGRD 업데이트
  UPDATE zte3fi0002
     SET stblg    = lv_belnr
         stjah    = gv_sel_gjahr
         stgrd    = gv_stgrd
         rev_date = sy-datum
         pernr    = gs_emp-pernr
  WHERE bukrs EQ gv_sel_bukrs
    AND gjahr EQ gv_sel_gjahr
    AND belnr EQ gv_sel_belnr.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH '원본 전표 업데이트 실패.' DISPLAY LIKE 'E'.
    ROLLBACK WORK. RETURN.
  ENDIF.

  COMMIT WORK AND WAIT.

  " 8. UI 데이터 갱신
  PERFORM refresh_save USING lv_belnr.

  MESSAGE s000 WITH lv_belnr '번 역분개 전표가 생성되었습니다.'.

ENDFORM.

*&---------------------------------------------------------------------*
FORM reverse_items  USING   pv_belnr TYPE zte3fi0002-belnr.

  CLEAR gt_rev_item.

  "역분개 로직
  LOOP AT gt_item INTO DATA(ls_orig).

    APPEND VALUE #( BASE CORRESPONDING #( ls_orig )
      belnr   = pv_belnr
      blart   = 'AB'
      shkzg   = ls_orig-shkzg
      dmbtr   = ls_orig-dmbtr * -1
      wrbtr   = ls_orig-wrbtr * -1
      dmbtr_s = COND #( WHEN ls_orig-shkzg = 'S'
                         THEN abs( ls_orig-dmbtr ) ELSE 0 )
      dmbtr_h = COND #( WHEN ls_orig-shkzg = 'H'
                         THEN abs( ls_orig-dmbtr ) ELSE 0 )
    ) TO gt_rev_item.

  ENDLOOP.

ENDFORM.

FORM set_domain_text .

  SELECT domvalue_l AS key, ddtext
    INTO TABLE @DATA(lt_stgrd)
    FROM dd07t

      READ TABLE lt_stgrd INTO DATA(ls_stgrd)

        <fs_h>-stgrd_txt = ls_stgrd-ddtext.

    ENDIF.

    IF <fs_h>-blart IS NOT INITIAL.
      READ TABLE lt_blart INTO DATA(ls_blart)
        WITH KEY key = <fs_h>-blart BINARY SEARCH.

      IF sy-subrc EQ 0.
        <fs_h>-blart_txt = ls_blart-ddtext.
      ENDIF.

    ENDIF.

  ENDLOOP.

*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM fill_voucher .

  DATA: lv_line      TYPE i,
        lv_val       TYPE char100,
        lv_sheet     TYPE i VALUE 1,
        lv_blart_txt TYPE dd07t-ddtext,
        lv_item_cnt  TYPE i,
        lv_del_from  TYPE i,
        ls_pagesetup TYPE ole2_object.

  CONSTANTS: lc_start_row    TYPE i VALUE 12.
  CONSTANTS: lc_template_end TYPE i VALUE 31.
  CONSTANTS: lc_max_rows     TYPE i VALUE 20.

  " 전표유형 도메인 텍스트
  SELECT SINGLE ddtext
    INTO lv_blart_txt
    FROM dd07t
   WHERE domname    EQ 'ZDE3_FI_BLART'
     AND domvalue_l EQ gs_header-blart
     AND ddlanguage EQ sy-langu.

  IF lv_blart_txt IS INITIAL.
    lv_blart_txt = gs_header-blart.
  ENDIF.

  " 헤더 매핑
  PERFORM fill_cells USING 6 3 'Masterpack'.
  PERFORM fill_cells USING 7 3 gs_header-gjahr.
  PERFORM fill_cells USING 8 3 lv_blart_txt.
  PERFORM fill_cells USING 9 3 gs_header-waers.
  PERFORM fill_cells USING 6 6 gs_header-belnr.

  lv_val = |{ gs_header-budat(4) }.{ gs_header-budat+4(2) }.{ gs_header-budat+6(2) }|.
  PERFORM fill_cells USING 7 6 lv_val.

  lv_val = |{ gs_header-bldat(4) }.{ gs_header-bldat+4(2) }.{ gs_header-bldat+6(2) }|.
  PERFORM fill_cells USING 8 6 lv_val.

  PERFORM fill_cells USING 9 6 gs_header-bltxt.

  lv_item_cnt = lines( gt_item_data ).

  " 데이터가 20행 초과 → 13행에 행 삽입
  IF lv_item_cnt > lc_max_rows.
    CALL METHOD OF gs_excel 'Rows' = gs_row_e
      EXPORTING #1 = 13.
    DO lv_item_cnt - lc_max_rows TIMES.
      CALL METHOD OF gs_row_e 'Insert'
        EXPORTING
          #1 = -4121.
    ENDDO.
  ENDIF.

  " 아이템 매핑
  lv_line = lc_start_row.

  LOOP AT gt_item_data INTO DATA(ls_item).
    PERFORM fill_cells USING lv_line 2 ls_item-saknr.
    PERFORM fill_cells USING lv_line 3 ls_item-saknr_txt.
    PERFORM fill_cells USING lv_line 6 'KRW'.

    CLEAR lv_val.
    " [수정] DMBTR은 항상 원화(KRW) 기준이므로 'KRW'로 하드코딩
    WRITE ls_item-dmbtr TO lv_val CURRENCY 'KRW'.
    CONDENSE lv_val.

    IF ls_item-shkzg = 'S'.
      PERFORM fill_cells USING lv_line 4 lv_val.
    ELSEIF ls_item-shkzg = 'H'.
      PERFORM fill_cells USING lv_line 5 lv_val.
    ENDIF.

    lv_line = lv_line + 1.
  ENDLOOP.

  " 데이터가 20행 미만 → 남은 빈 행 삭제
  IF lv_item_cnt < lc_max_rows.
    lv_del_from = lv_line.

    CALL METHOD OF gs_excel 'Cells' = gs_cell
      EXPORTING #1 = lv_del_from #2 = 1.
    CALL METHOD OF gs_excel 'Cells' = gs_cell2
      EXPORTING #1 = lc_template_end #2 = 1.
    CALL METHOD OF gs_excel 'Range' = gs_range
      EXPORTING #1 = gs_cell #2 = gs_cell2.
    GET PROPERTY OF gs_range 'EntireRow' = gs_row_e.
    CALL METHOD OF gs_row_e 'Delete'.
  ENDIF.

  " 합계행
  PERFORM fill_cells USING lv_line 6 'KRW'.

  CLEAR lv_val.
  " [수정] 합계 금액도 동일하게 'KRW' 기준 적용
  WRITE gv_popup_dmbtr_s TO lv_val CURRENCY 'KRW'. CONDENSE lv_val.
  PERFORM fill_cells USING lv_line 4 lv_val.

  CLEAR lv_val.
  WRITE gv_popup_dmbtr_h TO lv_val CURRENCY 'KRW'. CONDENSE lv_val.
  PERFORM fill_cells USING lv_line 5 lv_val.

  " 페이지 설정
  CALL METHOD OF gs_excel 'SHEETS' = gs_sheet
    EXPORTING #1 = 1.
  GET PROPERTY OF gs_sheet 'PageSetup' = ls_pagesetup.
  SET PROPERTY OF ls_pagesetup 'PrintTitleRows' = '$1:$11'.
  SET PROPERTY OF ls_pagesetup 'PaperSize'      = 9.
  SET PROPERTY OF ls_pagesetup 'Orientation'    = 1.
  SET PROPERTY OF ls_pagesetup 'FitToPagesWide' = 1.
  SET PROPERTY OF ls_pagesetup 'FitToPagesTall' = 0.
  SET PROPERTY OF ls_pagesetup 'Zoom'           = 100.

  DATA lv_footer_row TYPE i.
  lv_footer_row = lv_line + 2.

  lv_val = |출력일: { sy-datum(4) }.{ sy-datum+4(2) }.{ sy-datum+6(2) }  ·  Masterpack ERP System|.
  PERFORM fill_cells USING lv_footer_row 2 lv_val.

  FREE OBJECT: ls_pagesetup, gs_cell, gs_cell2, gs_range, gs_row_e.

  PERFORM make_pdf USING lv_sheet.
ENDFORM.

*&---------------------------------------------------------------------*
*& Form sum_amt
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM sum_amt .

  DATA: BEGIN OF ls_raw,
          belnr TYPE zte3fi0003-belnr,
          bukrs TYPE zte3fi0003-bukrs,
          gjahr TYPE zte3fi0003-gjahr,
          dmbtr TYPE zte3fi0003-dmbtr,
        END OF ls_raw.

  DATA lt_raw LIKE TABLE OF ls_raw.
  DATA lt_sum LIKE TABLE OF ls_raw.

  DATA lv_doc_amt TYPE p DECIMALS 2.

  CLEAR: gv_amt_total, gv_amt_pending, gv_amt_cleared,
         gv_amt_kunnr, gv_amt_kunnr_pnd,
         gv_amt_lifnr, gv_amt_lifnr_pnd.

  CHECK gt_header IS NOT INITIAL.

  "차변 아이템 금액 조회
  SELECT belnr bukrs gjahr dmbtr
    INTO CORRESPONDING FIELDS OF TABLE lt_raw
    FROM zte3fi0003
    FOR ALL ENTRIES IN gt_header
   WHERE bukrs EQ gt_header-bukrs
     AND gjahr EQ gt_header-gjahr
     AND belnr EQ gt_header-belnr
     AND shkzg EQ 'S'.

  " belnr+bukrs+gjahr 기준 dmbtr 합산
  LOOP AT lt_raw INTO ls_raw.
    COLLECT ls_raw INTO lt_sum.
  ENDLOOP.

  SORT lt_sum BY belnr bukrs gjahr.

  LOOP AT gt_header INTO DATA(ls_header).
    CLEAR lv_doc_amt.
    " 전표 합계 금액 취득 (없으면 0)
    lv_doc_amt = VALUE #( lt_sum[ belnr = ls_header-belnr
                                  bukrs = ls_header-bukrs
                                  gjahr = ls_header-gjahr ]-dmbtr OPTIONAL ).
    " 역분개(AB) / 역분개 대상 전표 제외
    CHECK ls_header-blart NE 'AB' AND ls_header-stblg EQ space.

    gv_amt_total += lv_doc_amt.
    " 미결 금액
    IF ls_header-bstat EQ space AND ls_header-open_flag EQ 'X'.
      gv_amt_pending += lv_doc_amt.
    ENDIF.
    " 반제 금액
    IF ls_header-bstat EQ 'C'.
      gv_amt_cleared += lv_doc_amt.
    ENDIF.
    " AR 금액
    IF ls_header-kunnr IS NOT INITIAL.
      gv_amt_kunnr += lv_doc_amt.
      IF ls_header-bstat EQ space AND ls_header-open_flag EQ 'X'.
        gv_amt_kunnr_pnd += lv_doc_amt.
      ENDIF.

*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM select_folder_path .

  CLEAR gv_pfolder.

  IF gcl_objfile IS NOT BOUND.
    CREATE OBJECT gcl_objfile.
  ENDIF.

  " 초기 폴더 설정
  IF gv_pfolder IS NOT INITIAL.
    gv_initialfolder = gv_pfolder.
  ELSE.
    gcl_objfile->get_temp_directory( CHANGING temp_dir = gv_initialfolder ).
  ENDIF.

  " 폴더 선택 전 변수 초기화 (취소 여부 확인용)
  CLEAR gv_pickedfolder.

  " 폴더 선택 팝업
  gcl_objfile->directory_browse(
    EXPORTING initial_folder  = gv_initialfolder
    CHANGING  selected_folder = gv_pickedfolder
    EXCEPTIONS OTHERS = 1 ).

  " sy-subrc가 0이더라도 경로가 없으면 취소로 간주
  IF sy-subrc = 0 AND gv_pickedfolder IS NOT INITIAL.
    gv_pfolder = gv_pickedfolder.
  ELSE.
    MESSAGE s000 WITH 'PDF 다운이 취소되었습니다.'.
  ENDIF.

  " 객체 해제
  IF gcl_objfile IS BOUND.
    gcl_objfile->free( ).
    FREE gcl_objfile.
  ENDIF.

ENDFORM.

FORM pdf_download .
*
*  DATA: lv_line        TYPE i,
*        lv_val         TYPE char100,
*        lv_sheet       TYPE i VALUE 1,
*        lv_total_dmbtr TYPE dmbtr,
*        lv_total_wrbtr TYPE wrbtr,
*        lv_sep         TYPE char1,

*        lv_data_cnt    TYPE i,

*        lv_insert_row  TYPE i,
*        lv_jpy_wrbtr   TYPE p DECIMALS 2.
*
*  " 폴더 선택
*  PERFORM select_folder_path.
*  CHECK gv_pfolder IS NOT INITIAL.
*
*  CALL METHOD cl_gui_frontend_services=>get_file_separator
*    CHANGING
*      file_separator = lv_sep.
*
*  gv_temp_filename     = |{ gv_pfolder }{ lv_sep }전표리스트_{ sy-datum }_{ sy-uzeit }.xlsx|.
*  gv_temp_filename_pdf = |{ gv_pfolder }{ lv_sep }전표리스트_{ sy-datum }_{ sy-uzeit }.pdf|.
*
*  PERFORM download_template USING 'ZPE3FI_POST_PDFALL2' gv_temp_filename.
*  PERFORM open_excel_template.
*
*  " 대량 데이터 대응을 위한 행 삽입
*  lv_data_cnt   = lines( gt_header ).
*  lv_insert_row = 8.
*
*  IF lv_data_cnt > 1.
*    CALL METHOD OF gs_excel 'Rows' = gs_row_e
*      EXPORTING #1 = lv_insert_row.
*    DO lv_data_cnt - 1 TIMES.
*      CALL METHOD OF gs_row_e 'Insert'
*        EXPORTING
*          #1 = -4121.
*    ENDDO.
*  ENDIF.
*
*  " 데이터 루프 (7행부터)
*  lv_line = 7.
*
*  LOOP AT gt_header INTO DATA(ls_out).
*
*    " 전표유형 도메인 텍스트
*    SELECT SINGLE ddtext
*      INTO lv_blart_txt
*      FROM dd07t
*     WHERE domname    EQ 'ZDE3_FI_BLART'
*       AND domvalue_l EQ ls_out-blart
*       AND ddlanguage EQ sy-langu.
*    IF lv_blart_txt IS INITIAL. lv_blart_txt = ls_out-blart. ENDIF.
*
*    PERFORM fill_cells USING lv_line 2 ls_out-belnr.
*    PERFORM fill_cells USING lv_line 3 ls_out-bukrs.
*    PERFORM fill_cells USING lv_line 4 ls_out-gjahr.
*    PERFORM fill_cells USING lv_line 5 lv_blart_txt.
*
*    lv_val = |{ ls_out-budat(4) }.{ ls_out-budat+4(2) }.{ ls_out-budat+6(2) }|.
*    PERFORM fill_cells USING lv_line 6 lv_val.
*

*    SELECT SUM( wrbtr )
*      INTO @lv_total_wrbtr
*      FROM zte3fi0003
*     WHERE belnr EQ @ls_out-belnr
*       AND bukrs EQ @ls_out-bukrs
*       AND gjahr EQ @ls_out-gjahr
*       AND shkzg EQ 'S'
*       AND fwaer NE 'KRW'.
*
*    " KRW 금액 - 차변만
*    SELECT SUM( dmbtr )
*      INTO @lv_total_dmbtr
*      FROM zte3fi0003
*     WHERE belnr EQ @ls_out-belnr
*       AND bukrs EQ @ls_out-bukrs
*       AND gjahr EQ @ls_out-gjahr
*       AND shkzg EQ 'S'.
*
*    IF lv_total_wrbtr <> 0.
*      CLEAR lv_val.
*      IF ls_out-waers = 'JPY'.
*        WRITE lv_total_wrbtr TO lv_val CURRENCY 'JPY'.
*      ELSE.
*        WRITE lv_total_wrbtr TO lv_val CURRENCY ls_out-waers.
*      ENDIF.
*      CONDENSE lv_val.
*      PERFORM fill_cells USING lv_line 7 lv_val.
*      PERFORM fill_cells USING lv_line 8 ls_out-waers.
*    ELSE.
*      PERFORM fill_cells USING lv_line 7 ''.
*      PERFORM fill_cells USING lv_line 8 ''.
*    ENDIF.
*
*    " KRW 금액
*    CLEAR lv_val.
*    WRITE lv_total_dmbtr TO lv_val CURRENCY 'KRW'. CONDENSE lv_val.
*    PERFORM fill_cells USING lv_line 9 lv_val.
*    PERFORM fill_cells USING lv_line 10 'KRW'.
*
*    PERFORM fill_cells USING lv_line 11 ls_out-bltxt.

*  CALL METHOD OF gs_excel 'Cells' = gs_cell2

*  CALL METHOD OF gs_excel 'Range' = gs_range
*    EXPORTING #1 = gs_cell #2 = gs_cell2.
*  GET PROPERTY OF gs_range 'EntireRow' = gs_row_e.
*  CALL METHOD OF gs_row_e 'Delete'.
*
*  " 출력일 footer
*  lv_footer_row = lv_line + 2.
*  lv_val = |출력일: { sy-datum(4) }.{ sy-datum+4(2) }.{ sy-datum+6(2) }  ·  Masterpack ERP System|.
*  PERFORM fill_cells USING lv_footer_row 2 lv_val.
*
*  " 페이지 설정
*  CALL METHOD OF gs_excel 'SHEETS' = gs_sheet
*    EXPORTING #1 = 1.
*  GET PROPERTY OF gs_sheet 'PageSetup' = ls_pagesetup.
*  SET PROPERTY OF ls_pagesetup 'PrintTitleRows' = '$1:$6'.
*  SET PROPERTY OF ls_pagesetup 'PaperSize'      = 9.
*  SET PROPERTY OF ls_pagesetup 'Orientation'    = 2.
*  SET PROPERTY OF ls_pagesetup 'Zoom'           = 85.
*
*  FREE OBJECT: ls_pagesetup, gs_cell, gs_cell2, gs_range, gs_row_e.
*
*  PERFORM make_pdf USING lv_sheet.
*
*  MESSAGE s000 WITH |총 { lv_data_cnt }건의 PDF 작업이 완료되었습니다.|.

  DATA: lv_line         TYPE i,
        lv_val          TYPE char100,
        lv_sheet        TYPE i VALUE 1,
        lv_total_dmbtr  TYPE dmbtr,
        lv_total_wrbtr  TYPE wrbtr,
        lv_sep          TYPE char1,
        ls_pagesetup    TYPE ole2_object,
        lv_data_cnt     TYPE i,
        lv_blart_txt    TYPE dd07t-ddtext,
        lv_footer_row   TYPE i,
        lv_insert_row   TYPE i,

        " [추가] 대량 Insert 및 삭제용 변수
        lv_end_row      TYPE i,
        lv_range_str    TYPE string,
        lv_template_max TYPE i VALUE 200. " 실제 템플릿의 여백 유지 가능 최대 라인수

  " 폴더 선택
  PERFORM select_folder_path.
  CHECK gv_pfolder IS NOT INITIAL.

  CALL METHOD cl_gui_frontend_services=>get_file_separator
    CHANGING
      file_separator = lv_sep.

  gv_temp_filename     = |{ gv_pfolder }{ lv_sep }전표리스트_{ sy-datum }_{ sy-uzeit }.xlsx|.
  gv_temp_filename_pdf = |{ gv_pfolder }{ lv_sep }전표리스트_{ sy-datum }_{ sy-uzeit }.pdf|.

  PERFORM download_template USING 'ZPE3FI_POST_PDFALL2' gv_temp_filename.
  PERFORM open_excel_template.

  lv_insert_row = 8.

  " [속도 개선] 대량 데이터 대응을 위한 행 삽입 (Range를 이용하여 한 번에 삽입)
  IF lv_data_cnt > 1.
    lv_end_row = lv_insert_row + lv_data_cnt - 2.

  " 데이터 루프 (7행부터)
  lv_line = 7.

  LOOP AT gt_header INTO DATA(ls_out).

    SELECT SINGLE ddtext

    SELECT SUM( wrbtr )
      INTO @lv_total_wrbtr
      FROM zte3fi0003
     WHERE belnr EQ @ls_out-belnr
       AND bukrs EQ @ls_out-bukrs
       AND gjahr EQ @ls_out-gjahr
       AND shkzg EQ 'S'

    " KRW 금액 - 차변만
    SELECT SUM( dmbtr )
      INTO @lv_total_dmbtr
      FROM zte3fi0003
     WHERE belnr EQ @ls_out-belnr
       AND bukrs EQ @ls_out-bukrs
       AND gjahr EQ @ls_out-gjahr
       AND shkzg EQ 'S'.

    " [수정] JPY 외화 표기 및 소수점 강제 치환 로직
    IF lv_total_wrbtr <> 0.
      CLEAR lv_val.
      IF ls_out-waers = 'JPY'.
        WRITE lv_total_wrbtr TO lv_val CURRENCY 'JPY' DECIMALS 0.
        CONDENSE lv_val.
        REPLACE ALL OCCURRENCES OF '.00' IN lv_val WITH ''.
        REPLACE ALL OCCURRENCES OF ',00' IN lv_val WITH ''.
      ELSE.
        WRITE lv_total_wrbtr TO lv_val CURRENCY ls_out-waers.
        CONDENSE lv_val.
      ENDIF.

      PERFORM fill_cells USING lv_line 7 lv_val.
      PERFORM fill_cells USING lv_line 8 ls_out-waers.
    ELSE.
      PERFORM fill_cells USING lv_line 7 ''.
      PERFORM fill_cells USING lv_line 8 ''.
    ENDIF.

    " KRW 금액
    CLEAR lv_val.
    WRITE lv_total_dmbtr TO lv_val CURRENCY 'KRW'. CONDENSE lv_val.
    PERFORM fill_cells USING lv_line 9 lv_val.
    PERFORM fill_cells USING lv_line 10 'KRW'.

    PERFORM fill_cells USING lv_line 11 ls_out-bltxt.

    lv_line = lv_line + 1.
  ENDLOOP.

  " [서식 보호] 잔여 행 삭제 (템플릿 한계치 내에서만 삭제하여 하단 여백 및 서식 유지)
  IF lv_line < lv_template_max.
    CALL METHOD OF gs_excel 'Cells' = gs_cell
      EXPORTING #1 = lv_line #2 = 1.
    CALL METHOD OF gs_excel 'Cells' = gs_cell2
      EXPORTING #1 = lv_template_max #2 = 1.
    CALL METHOD OF gs_excel 'Range' = gs_range
      EXPORTING #1 = gs_cell #2 = gs_cell2.
    GET PROPERTY OF gs_range 'EntireRow' = gs_row_e.
    CALL METHOD OF gs_row_e 'Delete'.
  ENDIF.

  " 출력일 footer
  lv_footer_row = lv_line + 2.
  lv_val = |출력일: { sy-datum(4) }.{ sy-datum+4(2) }.{ sy-datum+6(2) }  ·  Masterpack ERP System|.
  PERFORM fill_cells USING lv_footer_row 2 lv_val.

  " 페이지 설정
  CALL METHOD OF gs_excel 'SHEETS' = gs_sheet
    EXPORTING #1 = 1.
  GET PROPERTY OF gs_sheet 'PageSetup' = ls_pagesetup.
  SET PROPERTY OF ls_pagesetup 'PrintTitleRows' = '$1:$6'.
  SET PROPERTY OF ls_pagesetup 'PaperSize'      = 9.
  SET PROPERTY OF ls_pagesetup 'Orientation'    = 2.
  SET PROPERTY OF ls_pagesetup 'Zoom'           = 85.

  FREE OBJECT: ls_pagesetup, gs_cell, gs_cell2, gs_range, gs_row_e.

  PERFORM make_pdf USING lv_sheet.

  MESSAGE s000 WITH |총 { lv_data_cnt }건의 PDF 작업이 완료되었습니다.|.

ENDFORM.

*& Form check_bp
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM check_bp .

  DATA: lv_gjahr_tmp  TYPE zte3fi0002-gjahr,
        lv_budat_low  TYPE zte3fi0002-budat,
        lv_budat_high TYPE zte3fi0002-budat.

  CASE tab_block-activetab.
    WHEN 'COMM1'.
      lv_gjahr_tmp  = pa_gjahr.
      lv_budat_low  = so_budat-low.
      lv_budat_high = so_budat-high.

    WHEN 'COMM2'. " 고객
      lv_gjahr_tmp  = pa_gja2.
      lv_budat_low  = so_bud2-low.
      lv_budat_high = so_bud2-high.

      " 시작값(Low) 체크
      IF so_kun-low IS NOT INITIAL.
        SELECT SINGLE kunnr FROM zte3sd0001 INTO @DATA(lv_k_low) WHERE kunnr = @so_kun-low.
        IF sy-subrc NE 0.
          MESSAGE s000 WITH '잘못된 고객번호입니다.' DISPLAY LIKE 'E'.
          STOP.
        ENDIF.
      ENDIF.
      " 종료값(High) 체크
      IF so_kun-high IS NOT INITIAL.
        SELECT SINGLE kunnr FROM zte3sd0001 INTO @DATA(lv_k_high) WHERE kunnr = @so_kun-high.
        IF sy-subrc <> 0.
          MESSAGE s000 WITH '잘못된 고객번호입니다.' DISPLAY LIKE 'E'.
          STOP.
        ENDIF.
      ENDIF.

    WHEN 'COMM3'. " 공급업체
      lv_gjahr_tmp = pa_gja3.
      lv_budat_low = so_bud3-low.
      lv_budat_high = so_bud3-high.

      " 시작값(Low) 체크
      IF so_lif-low IS NOT INITIAL.
        SELECT SINGLE lifnr FROM zte3mm0002 INTO @DATA(lv_l_low) WHERE lifnr = @so_lif-low.
        IF sy-subrc NE 0.
          MESSAGE s000 WITH '잘못된 공급업체번호입니다.' DISPLAY LIKE 'E'.
          STOP.
        ENDIF.
      ENDIF.
      " 종료값(High) 체크
      IF so_lif-high IS NOT INITIAL.
        SELECT SINGLE lifnr FROM zte3mm0002 INTO @DATA(lv_l_high) WHERE lifnr = @so_lif-high.
        IF sy-subrc NE 0.
          MESSAGE s000 WITH '잘못된 공급업체번호입니다.' DISPLAY LIKE 'E'.
          STOP.
        ENDIF.
      ENDIF.
  ENDCASE.

  IF gv_min_gjahr IS INITIAL.
    gv_min_gjahr = sy-datum(4).
  ENDIF.

  IF lv_gjahr_tmp IS INITIAL.
    MESSAGE s000 WITH '회계연도를 입력하세요.' DISPLAY LIKE 'E'.
    STOP.

  ELSEIF lv_gjahr_tmp < gv_min_gjahr OR lv_gjahr_tmp > sy-datum(4).
    MESSAGE s000 WITH |{ gv_min_gjahr }년부터 현재까지만 조회 가능합니다.| DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  IF lv_budat_low IS NOT INITIAL.
    " 기간 역전 체크
    IF lv_budat_high IS NOT INITIAL AND lv_budat_low > lv_budat_high.
      MESSAGE s000 WITH '시작일이 종료일보다 클 수 없습니다.' DISPLAY LIKE 'E'.
      STOP.
    ENDIF.
    " 연도 일치 체크 (Low와 High 둘 다 체크)
    IF lv_budat_low(4) NE lv_gjahr_tmp OR
       ( lv_budat_high IS NOT INITIAL AND lv_budat_high(4) NE lv_gjahr_tmp ).
      MESSAGE s000 WITH '전기일의 연도가 회계연도와 일치하지 않습니다.' DISPLAY LIKE 'E'.
      STOP.
    ENDIF.
    " 오늘 날짜 체크
    IF lv_budat_low > sy-datum.
      MESSAGE s000 WITH '오늘 이후 날짜는 조회할 수 없습니다.' DISPLAY LIKE 'E'.
      STOP.
    ENDIF.
  ENDIF.

ENDFORM.
