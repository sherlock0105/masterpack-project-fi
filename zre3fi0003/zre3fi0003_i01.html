MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'REFRESH'.
      CLEAR gv_okcode.
      CLEAR gv_mode.

      PERFORM get_header_data.
      PERFORM set_sort.
      PERFORM set_style.
      PERFORM refresh_grid.

      MESSAGE s000 WITH '새로고침 되었습니다.'.

    WHEN 'PDF'.
      CLEAR gv_okcode.
      PERFORM pdf_down_all.

    WHEN 'AR'.
      CLEAR gv_okcode.
      PERFORM go_ar.

    WHEN 'AP'.
      CLEAR gv_okcode.
      PERFORM go_ap.

    WHEN OTHERS.

  ENDCASE.

ENDMODULE.
