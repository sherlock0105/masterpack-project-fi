MODULE exit_0100 INPUT.

  IF gcl_grid_sim IS BOUND.
    gcl_grid_sim->refresh_table_display(
      EXPORTING
        i_soft_refresh = space
        is_stable = gs_stable ).
  ENDIF.

  IF gcl_tree IS BOUND.
    gcl_tree->frontend_update( ).
  ENDIF.

  IF gcl_grid_hold IS BOUND.
    gcl_grid_hold->refresh_table_display(
      EXPORTING
        is_stable      = gs_stable                 " With Stable Rows/Columns
        i_soft_refresh = space ) .            " Without Sort, Filter, etc.
  ENDIF.

  LEAVE TO SCREEN 0.

ENDMODULE.
