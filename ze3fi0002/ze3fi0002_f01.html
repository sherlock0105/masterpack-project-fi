FORM filter_detail USING pv_node_key TYPE lvc_nkey.

  DATA ls_node LIKE gs_data.

  CALL METHOD gcl_tree->get_outtab_line
    EXPORTING
      i_node_key    = pv_node_key
    IMPORTING
      e_outtab_line = ls_node.

  REFRESH gt_data_disp.

  CASE pv_node_key.
    WHEN gv_all_node.
      LOOP AT gt_data INTO gs_data WHERE belnr IS NOT INITIAL.
        APPEND gs_data TO gt_data_disp.
      ENDLOOP.

    WHEN gv_ar_node.
      LOOP AT gt_data INTO gs_data WHERE ar_ap = 'AR' AND belnr IS NOT INITIAL.
        APPEND gs_data TO gt_data_disp.
        gv_check_node = 'AR'.
      ENDLOOP.

    WHEN gv_ap_node.
      LOOP AT gt_data INTO gs_data WHERE ar_ap = 'AP' AND belnr IS NOT INITIAL.
        APPEND gs_data TO gt_data_disp.
        gv_check_node = 'AP'.
      ENDLOOP.

    WHEN OTHERS.
      IF ls_node-belnr IS NOT INITIAL.
        LOOP AT gt_data INTO gs_data WHERE belnr = ls_node-belnr.
          APPEND gs_data TO gt_data_disp.
        ENDLOOP.

      ELSEIF ls_node-name1 IS NOT INITIAL.
        LOOP AT gt_data INTO gs_data
          WHERE ar_ap = 'AR'
            AND name1 = ls_node-name1
            AND fwaer = ls_node-fwaer
            AND belnr IS NOT INITIAL.
          APPEND gs_data TO gt_data_disp.
          gv_check_node = 'AR'.
        ENDLOOP.

      ELSEIF ls_node-vennm IS NOT INITIAL.
        LOOP AT gt_data INTO gs_data
          WHERE ar_ap = 'AP'
            AND vennm = ls_node-vennm
            AND fwaer = ls_node-fwaer
            AND belnr IS NOT INITIAL.
          APPEND gs_data TO gt_data_disp.
          gv_check_node = 'AP'.
        ENDLOOP.

      ELSEIF ls_node-fwaer IS NOT INITIAL.
        LOOP AT gt_data INTO gs_data
          WHERE ar_ap = ls_node-ar_ap
            AND fwaer = ls_node-fwaer
            AND belnr IS NOT INITIAL.
          APPEND gs_data TO gt_data_disp.
        ENDLOOP.

      ENDIF.

  ENDCASE.

  SORT gt_data_disp BY ar_ap dmbtr DESCENDING belnr fwaer.

  gcl_grid_sim->refresh_table_display(
    EXPORTING
      is_stable      = gs_stable
      i_soft_refresh = space ).




ENDFORM.

    FROM zcds_e3_fi_0001(
           p_usd = @gv_usd,
           p_eur = @gv_eur,
           p_jpy = @gv_jpy
         )
   WHERE budat <= @gv_evdat
     AND bukrs = @gv_bukrs
*     AND gjahr = ''
     AND  shkzg  =  'S'
     AND ( kunnr NE @space OR lifnr NE @space )
    INTO CORRESPONDING FIELDS OF TABLE @gt_data.


  IF gt_data IS INITIAL.
    MESSAGE s000 WITH '미결 외화전표가 없습니다.'.
    RETURN.
  ENDIF.

  SORT gt_data BY ar_ap dmbtr DESCENDING belnr fwaer.

  MOVE-CORRESPONDING gt_data TO gt_data_disp.

  SELECT SINGLE butxt
    FROM zte3cm0001
    INTO gv_butxt
   WHERE bukrs = gv_bukrs.

FORM set_chart .

  CHECK gcl_chart IS BOUND.

  DATA : BEGIN OF gs_top,
           name  TYPE string,
           dmbtr TYPE p DECIMALS 2,
           ar_ap TYPE c LENGTH 2,
         END OF gs_top.

  DATA: lt_ar    LIKE TABLE OF gs_top,
        lt_ap    LIKE TABLE OF gs_top,
        ls_top   LIKE LINE OF lt_ar,
        lv_xdata TYPE xstring,
        lv_val   TYPE string.

        lv_cats  TYPE REF TO if_ixml_element,
        lv_cat   TYPE REF TO if_ixml_element,
        lv_ser   TYPE REF TO if_ixml_element,
        lv_point TYPE REF TO if_ixml_element,
        lv_value TYPE REF TO if_ixml_element.

  DATA: lcl_ixml    TYPE REF TO if_ixml,
        lcl_sf      TYPE REF TO if_ixml_stream_factory,
        lcl_enc     TYPE REF TO if_ixml_encoding,
        lv_cust_doc TYPE REF TO if_ixml_document,
        lv_cust_os  TYPE REF TO if_ixml_ostream,
        lv_cust_str TYPE xstring,
        lv_data_doc TYPE REF TO if_ixml_document,
        lv_data_os  TYPE REF TO if_ixml_ostream,
        lv_croot    TYPE REF TO if_ixml_element,
        lv_gs       TYPE REF TO if_ixml_element,
        lv_def      TYPE REF TO if_ixml_element,
        lv_vals     TYPE REF TO if_ixml_element,
        lv_elems    TYPE REF TO if_ixml_element,
        lv_celems   TYPE REF TO if_ixml_element,
        lv_title    TYPE REF TO if_ixml_element,
        lv_ttext    TYPE REF TO if_ixml_element,
        lv_pt       TYPE REF TO if_ixml_element.

  "======================================================
  " AR Top5 집계
  "======================================================
  LOOP AT gt_data INTO DATA(ls) WHERE ar_ap = 'AR' AND belnr IS NOT INITIAL.
    READ TABLE lt_ar ASSIGNING FIELD-SYMBOL(<fa>)
      WITH KEY name = CONV string( ls-name1 ).
    IF sy-subrc <> 0.
      CLEAR ls_top.
      ls_top-name  = ls-name1.
      ls_top-ar_ap = 'AR'.
      APPEND ls_top TO lt_ar.
      READ TABLE lt_ar ASSIGNING <fa> WITH KEY name = CONV string( ls-name1 ).
    ENDIF.
    <fa>-dmbtr += ls-dmbtr.
  ENDLOOP.

  "======================================================
  " AP Top5 집계
  "======================================================
  LOOP AT gt_data INTO ls WHERE ar_ap = 'AP' AND belnr IS NOT INITIAL.
    READ TABLE lt_ap ASSIGNING FIELD-SYMBOL(<fp>)
      WITH KEY name = CONV string( ls-vennm ).
    IF sy-subrc <> 0.
      CLEAR ls_top.
      ls_top-name  = ls-vennm.
      ls_top-ar_ap = 'AP'.
      APPEND ls_top TO lt_ap.
      READ TABLE lt_ap ASSIGNING <fp> WITH KEY name = CONV string( ls-vennm ).
    ENDIF.
    <fp>-dmbtr += ls-dmbtr.
  ENDLOOP.

  SORT lt_ar BY dmbtr DESCENDING.
  SORT lt_ap BY dmbtr DESCENDING.
  DELETE lt_ar FROM 6.
  DELETE lt_ap FROM 6.

  IF lt_ar IS INITIAL AND lt_ap IS INITIAL. RETURN. ENDIF.

  "======================================================
  lcl_ixml = cl_ixml=>create( ).

  lcl_enc  = lcl_ixml->create_encoding(
               byte_order    = if_ixml_encoding=>co_little_endian
               character_set = 'utf-8' ).

  "======================================================
  " 1. Customizing XML  (3D + Title + Point 색상)
  "======================================================
  lv_cust_doc = lcl_ixml->create_document( ).
  lv_cust_doc->set_encoding( lcl_enc ).

  lv_croot = lv_cust_doc->create_simple_element(
               name = 'SAPChartCustomizing' parent = lv_cust_doc ).
  lv_croot->set_attribute( name = 'version' value = '1.1' ).

  " GlobalSettings
  lv_gs = lv_cust_doc->create_simple_element(

  lv_cust_doc->create_simple_element( parent = lv_gs

*&---------------------------------------------------------------------*
*& Form build_tree_section
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> P_
*&      --> P_
*&      --> LV_ALL_KEY
*&      --> GV_AR_NODE
*&      --> LT_TEMP
*&---------------------------------------------------------------------*
FORM build_tree_section USING pv_type  TYPE c        "AP, AP
                              pv_text  TYPE string   "폴더 이름
                              pv_p_key TYPE lvc_nkey "부모 노드키
                              pv_key   TYPE lvc_nkey "출력 헤더 키 (AP, AP)
                              pt_data  LIKE gt_data.

  DATA: lv_curr_key TYPE lvc_nkey,
        lv_ptnr_key TYPE lvc_nkey,
        lv_node_key TYPE lvc_nkey.

  DATA: lt_curr    TYPE TABLE OF waers,
        lv_curr    TYPE waers,
        lv_ptnr_nm TYPE string,
        lv_prev_nm TYPE string.

  DATA: ls_node  LIKE gs_data,
        lt_local LIKE gt_data.   " 로컬 정렬용

  REFRESH: gt_curr, gt_curr_sum, gt_ptnr_sum.

  " 로컬 복사 후 거래처명 기준 정렬
  lt_local = pt_data.
  IF pv_type = 'AR'.
    SORT lt_local BY fwaer kunnr name1.
  ELSE.
    SORT lt_local BY fwaer lifnr vennm.
  ENDIF.

  LOOP AT lt_local INTO gs_data WHERE ar_ap = pv_type.
    CLEAR gs_curr_sum.
    gs_curr_sum-fwaer = gs_data-fwaer.
    gs_curr_sum-wrbtr = gs_data-wrbtr.
    COLLECT gs_curr_sum INTO gt_curr_sum.

    CLEAR gs_ptnr_sum.
    gs_ptnr_sum-fwaer = gs_data-fwaer.
    gs_ptnr_sum-ptnr  = COND #( WHEN pv_type = 'AR'
                                THEN gs_data-name1 ELSE gs_data-vennm ).
    gs_ptnr_sum-wrbtr = gs_data-wrbtr.
    COLLECT gs_ptnr_sum INTO gt_ptnr_sum.

    APPEND gs_data-fwaer TO lt_curr.
  ENDLOOP.

  SORT lt_curr.
  DELETE ADJACENT DUPLICATES FROM lt_curr.

  CLEAR ls_node.
  PERFORM add_node USING ls_node pv_text pv_p_key pv_key.

  LOOP AT lt_curr INTO lv_curr.
    CLEAR ls_node.
    ls_node-ar_ap = pv_type.
    ls_node-fwaer = lv_curr.

    READ TABLE gt_curr_sum INTO gs_curr_sum WITH KEY fwaer = lv_curr.
    IF sy-subrc = 0.
      ls_node-wrbtr = gs_curr_sum-wrbtr.
    ENDIF.

    PERFORM add_node USING ls_node lv_curr pv_key lv_curr_key.

    CLEAR lv_prev_nm.

    LOOP AT lt_local INTO gs_data WHERE ar_ap = pv_type AND fwaer = lv_curr.
      lv_ptnr_nm = COND #( WHEN pv_type = 'AR' THEN gs_data-name1 ELSE gs_data-vennm ).

      IF lv_ptnr_nm NE lv_prev_nm.
        CLEAR ls_node.
        ls_node-ar_ap = pv_type.
        ls_node-fwaer = lv_curr.
        IF pv_type = 'AR'. ls_node-name1 = lv_ptnr_nm. ELSE. ls_node-vennm = lv_ptnr_nm. ENDIF.

        READ TABLE gt_ptnr_sum INTO gs_ptnr_sum WITH KEY fwaer = lv_curr ptnr = lv_ptnr_nm.
        IF sy-subrc = 0.
          ls_node-wrbtr = gs_ptnr_sum-wrbtr.
        ENDIF.

        PERFORM add_node USING ls_node lv_ptnr_nm lv_curr_key lv_ptnr_key.
        lv_prev_nm = lv_ptnr_nm.
      ENDIF.

      PERFORM add_node USING gs_data gs_data-belnr lv_ptnr_key lv_node_key.
    ENDLOOP.
  ENDLOOP.

ENDFORM.

*&---------------------------------------------------------------------*
*& Form build_eval
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> GV_CHECK
*&---------------------------------------------------------------------*
FORM build_eval  USING pv_check.

  DATA: ls_header     TYPE zte3fi0002,
        lt_items      TYPE TABLE OF zte3fi0003,
        ls_item       TYPE zte3fi0003,
        lv_belnr      TYPE zte3fi0002-belnr,
        ls_rev_header TYPE zte3fi0002,
        lt_rev_items  TYPE TABLE OF zte3fi0003,
        lv_rev_belnr  TYPE zte3fi0002-belnr,
        lv_rev_date   TYPE sy-datum.

  DATA: lv_ar_diff TYPE p DECIMALS 2,
        lv_ap_diff TYPE p DECIMALS 2.


  LOOP AT gt_data INTO DATA(ls_data) WHERE belnr IS NOT INITIAL.
    IF ls_data-ar_ap = 'AR'.
      lv_ar_diff += ls_data-eval_diff.
    ELSEIF ls_data-ar_ap = 'AP'.
      lv_ap_diff += ls_data-eval_diff.
    ENDIF.
  ENDLOOP.

  CHECK lv_ar_diff NE 0 OR lv_ap_diff NE 0.

  REFRESH lt_items.
  " 헤더 세팅
  CLEAR ls_header.
  ls_header-bukrs = '1000'.
  ls_header-gjahr = gv_evdat(4).
  ls_header-blart = 'SA'.
  ls_header-budat = gv_evdat.
  ls_header-bldat = gv_evdat.
  ls_header-monat = gv_evdat+4(2).
  ls_header-waers = 'KRW'.
  ls_header-bltxt = |{ gv_evdat(4) }년 { gv_evdat+4(2) }월 외화환산평가|.

  " 1. AR (외화채권) 전표 라인 생성
  IF lv_ar_diff NE 0.
    CLEAR ls_item.
    ls_item-buzei = 1.
    ls_item-saknr = '1100000001'. " AR 채권 계정
    ls_item-koart = 'D'.
    ls_item-waers = 'KRW'.
    ls_item-dmbtr = abs( lv_ar_diff ).
    ls_item-shkzg = COND #( WHEN lv_ar_diff > 0 THEN 'S' ELSE 'H' ).
    APPEND ls_item TO lt_items.

    CLEAR ls_item.
    ls_item-buzei = 2.
    ls_item-saknr = COND #( WHEN lv_ar_diff > 0 THEN '4200000003' ELSE '5300000003' ).
    ls_item-koart = 'S'.
    ls_item-waers = 'KRW'.
    ls_item-kostl = 'CC_FIN'.
    ls_item-dmbtr = abs( lv_ar_diff ).
    ls_item-shkzg = COND #( WHEN lv_ar_diff > 0 THEN 'H' ELSE 'S' ).
    APPEND ls_item TO lt_items.
  ENDIF.

  " 2. AP (외화채무) 전표 라인 생성
  IF lv_ap_diff NE 0.
    DATA(lv_idx) = lines( lt_items ).
    CLEAR ls_item.
    ls_item-buzei = lv_idx + 1.
    ls_item-saknr = '2100000001'. " AP 채무 계정
    ls_item-koart = 'K'.
    ls_item-waers = 'KRW'.
    ls_item-dmbtr = abs( lv_ap_diff ).
    ls_item-shkzg = COND #( WHEN lv_ap_diff > 0 THEN 'S' ELSE 'H' ).
    APPEND ls_item TO lt_items.

    CLEAR ls_item.
    ls_item-buzei = lv_idx + 2.
    ls_item-saknr = COND #( WHEN lv_ap_diff > 0 THEN '4200000003' ELSE '5300000003' ).
    ls_item-koart = 'S'.
    ls_item-waers = 'KRW'.
    ls_item-kostl = 'CC_FIN'.
    ls_item-dmbtr = abs( lv_ap_diff ).
    ls_item-shkzg = COND #( WHEN lv_ap_diff > 0 THEN 'H' ELSE 'S' ).
    APPEND ls_item TO lt_items.
  ENDIF.

  " 실제 전표 생성
  IF pv_check NE 'X'.
    " 말일 평가 전표 생성
    zcl_e3_fi_document=>post_document(
      EXPORTING is_header = ls_header
                it_items  = lt_items
      IMPORTING ev_belnr  = lv_belnr ).

    IF lv_belnr IS NOT INITIAL.
      " 익월 1일 자 환입 전표 자동 생성
      lv_rev_date         = gv_evdat + 1.
      ls_rev_header       = ls_header.
      ls_rev_header-gjahr = lv_rev_date(4).
      ls_rev_header-budat = lv_rev_date.
      ls_rev_header-bldat = lv_rev_date.
      ls_rev_header-monat = lv_rev_date+4(2).
      ls_rev_header-bltxt = |{ gv_evdat(4) }년 { gv_evdat+4(2) }월 외화환산환입|.
      ls_rev_header-xblnr = lv_belnr. " 참조 필드에 평가 전표번호

      " 환입 아이템 복사
      lt_rev_items = lt_items.

      LOOP AT lt_rev_items ASSIGNING FIELD-SYMBOL(<fs_rev>).
        <fs_rev>-shkzg = COND #( WHEN <fs_rev>-shkzg = 'S' THEN 'H' ELSE 'S' ).

        IF <fs_rev>-koart = 'S'.
          <fs_rev>-saknr = COND #( WHEN <fs_rev>-saknr = '4200000003' THEN '5300000003'
                                   WHEN <fs_rev>-saknr = '5300000003' THEN '4200000003'
                                   ELSE <fs_rev>-saknr ).
        ENDIF.
      ENDLOOP.

      zcl_e3_fi_document=>post_document(
        EXPORTING is_header = ls_rev_header
                  it_items  = lt_rev_items
        IMPORTING ev_belnr  = lv_rev_belnr ).

      IF lv_rev_belnr IS NOT INITIAL.
        MESSAGE s000 WITH |평가 전표({ lv_belnr }) / 환입 전표({ lv_rev_belnr }) 생성되었습니다.|.
      ENDIF.

    ELSE.
      MESSAGE e000 WITH '평가전표 생성 실패.'.
    ENDIF.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_init_data .

  DATA(lv_gdatu) = 99999999 - gv_evdat.

  SELECT saknr stext
    INTO TABLE gt_saknr
    FROM zte3fi0001
   WHERE bukrs = gv_bukrs.

  SORT gt_saknr BY saknr.

  SELECT kunnr name1
    FROM zte3sd0001
    INTO TABLE gt_kunnr.

  SORT gt_kunnr BY kunnr.

  SELECT lifnr vennm
    FROM zte3mm0002
    INTO TABLE gt_lifnr.
  SORT gt_lifnr BY lifnr.

  SELECT DISTINCT fcurr gdatu ukurs ffact erdat
    INTO TABLE gt_rate
    FROM zte3fi0008
   WHERE tcurr = 'KRW'
     AND fcurr IN ( 'USD', 'EUR', 'JPY' )  " ← 필요한 통화만
     AND gdatu >= lv_gdatu.

  " gdatu 오름차순 = 숫자 작은 것 = 가장 최신 유효일자
  SORT gt_rate BY fcurr gdatu ASCENDING.
  DELETE ADJACENT DUPLICATES FROM gt_rate COMPARING fcurr.
ENDFORM.

***
  DATA : ls_color TYPE lvc_s_scol,

  DATA : lv_sum_usd TYPE p LENGTH 16 DECIMALS 2,

      WITH KEY fcurr = <fs_hold>-fwaer BINARY SEARCH.

                      WHEN sy-subrc = 0 AND ls_rate-ffact > 0
                      THEN ls_rate-ffact ELSE 1 ).

    DATA(lv_ukurs) = COND zte3fi0008-ukurs(
      WHEN sy-subrc = 0 THEN

          WHEN 'USD' THEN COND #( WHEN gv_sim_usd > 0 THEN gv_sim_usd ELSE ls_rate-ukurs )
          WHEN 'EUR' THEN COND #( WHEN gv_sim_eur > 0 THEN gv_sim_eur ELSE ls_rate-ukurs )

      ELSE 1 ).

    <fs_hold>-ukurs     = lv_ukurs.

    " 색상 처리

      ls_color-fname = 'UKURS'.     APPEND ls_color TO <fs_hold>-color.

    CASE <fs_hold>-fwaer.

      WHEN 'EUR'.

      WHEN 'JPY'.

*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_prev_data .
*
*  REFRESH gt_prev.
*
*  DATA: lv_seq      TYPE i,
*        lv_rev_date TYPE sy-datum,
*        lt_items    TYPE TABLE OF zte3fi0003,
*        ls_item     TYPE zte3fi0003,
*        lt_rev      TYPE TABLE OF zte3fi0003.
*
*  DATA: lv_total_diff TYPE p DECIMALS 2.
*
*  LOOP AT gt_data_disp INTO gs_data WHERE belnr IS NOT INITIAL.
*    lv_total_diff += gs_data-eval_diff.
*  ENDLOOP.
*
*  " 평가 대상 금액이 없으면 전표 생성 불가
*  CHECK lv_total_diff NE 0.


  REFRESH gt_prev.

  DATA: lv_seq      TYPE i,
        lv_rev_date TYPE sy-datum,
        lt_items    TYPE TABLE OF zte3fi0003,
        ls_item     TYPE zte3fi0003,
        lt_rev      TYPE TABLE OF zte3fi0003.

  DATA: lv_total_diff TYPE p DECIMALS 2.

  " ★ 수정: gt_data_disp(트리 필터링된 화면용) → gt_data(전체 원본)로 변경
  LOOP AT gt_data INTO gs_data WHERE belnr IS NOT INITIAL.
    lv_total_diff += gs_data-eval_diff.
  ENDLOOP.

  " 평가 대상 금액이 없으면 전표 생성 불가
  CHECK lv_total_diff NE 0.


  lv_rev_date = gv_evdat + 1.

  CLEAR ls_item.
  ls_item-buzei = 1.
  ls_item-saknr = '2100000001'. " 예시: 계정번호
  ls_item-koart = 'K'.
  ls_item-waers = 'KRW'.
  ls_item-dmbtr = abs( lv_total_diff ).
  ls_item-shkzg = COND #( WHEN lv_total_diff < 0 THEN 'H' ELSE 'S' ).
  APPEND ls_item TO lt_items.

  CLEAR ls_item.
  ls_item-buzei = 2.
  ls_item-saknr = COND #( WHEN lv_total_diff < 0 THEN '5300000003' ELSE '4200000003' ).
  ls_item-koart = 'S'.
  ls_item-waers = 'KRW'.
  ls_item-dmbtr = abs( lv_total_diff ).
  ls_item-shkzg = COND #( WHEN lv_total_diff < 0 THEN 'S' ELSE 'H' ).
  ls_item-kostl = 'CC_FIN'.
  APPEND ls_item TO lt_items.

  "환입전표
  lt_rev = lt_items.
  LOOP AT lt_rev ASSIGNING FIELD-SYMBOL(<fs>).
    <fs>-shkzg = COND #( WHEN <fs>-shkzg = 'S' THEN 'H' ELSE 'S' ).
    IF <fs>-koart = 'S'.
      <fs>-saknr = COND #( WHEN <fs>-saknr = '4200000003' THEN '5300000003'
                           WHEN <fs>-saknr = '5300000003' THEN '4200000003'
                           ELSE <fs>-saknr ).
    ENDIF.
  ENDLOOP.

  "평가 전표
  LOOP AT lt_items INTO ls_item.
    lv_seq += 1.
    CLEAR gs_prev.
    gs_prev-seq     = lv_seq.
    gs_prev-remark  = '평가'.
    gs_prev-budat   = gv_evdat.
    gs_prev-blart   = 'SA'.
    gs_prev-bltxt   = |{ gv_evdat(4) }년 { gv_evdat+4(2) }월 외화환산평가|.
    gs_prev-saknr   = ls_item-saknr.
    gs_prev-stext   = VALUE #( gt_saknr[ saknr = ls_item-saknr ]-stext OPTIONAL ).
    gs_prev-kostl   = ls_item-kostl.
    gs_prev-shkzg   = ls_item-shkzg.
    gs_prev-dmbtr_s = COND #( WHEN ls_item-shkzg = 'S' THEN ls_item-dmbtr ELSE 0 ).
    gs_prev-dmbtr_h = COND #( WHEN ls_item-shkzg = 'H' THEN ls_item-dmbtr ELSE 0 ).
    gs_prev-waers_s = COND #( WHEN ls_item-shkzg = 'S' THEN ls_item-waers ELSE space ).
    gs_prev-waers_h = COND #( WHEN ls_item-shkzg = 'H' THEN ls_item-waers ELSE space ).
    APPEND gs_prev TO gt_prev.
  ENDLOOP.

  "환입 전표
  LOOP AT lt_rev INTO ls_item.
    lv_seq += 1.
    CLEAR gs_prev.
    gs_prev-seq     = lv_seq.
    gs_prev-remark  = '환입'.
    gs_prev-budat   = lv_rev_date.
    gs_prev-blart   = 'SA'.
    gs_prev-bltxt   = |{ gv_evdat(4) }년 { gv_evdat+4(2) }월 외화환산환입|.
    gs_prev-saknr   = ls_item-saknr.
    gs_prev-stext   = VALUE #( gt_saknr[ saknr = ls_item-saknr ]-stext OPTIONAL ).
    gs_prev-kostl   = ls_item-kostl.
    gs_prev-shkzg   = ls_item-shkzg.
    gs_prev-dmbtr_s = COND #( WHEN ls_item-shkzg = 'S' THEN ls_item-dmbtr ELSE 0 ).
    gs_prev-dmbtr_h = COND #( WHEN ls_item-shkzg = 'H' THEN ls_item-dmbtr ELSE 0 ).
    gs_prev-waers_s = COND #( WHEN ls_item-shkzg = 'S' THEN ls_item-waers ELSE space ).
    gs_prev-waers_h = COND #( WHEN ls_item-shkzg = 'H' THEN ls_item-waers ELSE space ).
    APPEND gs_prev TO gt_prev.
  ENDLOOP.

  SORT gt_prev BY remark ASCENDING seq.
  PERFORM set_prev_color.

ENDFORM.

*&---------------------------------------------------------------------*
*& Form get_hist_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_hist_data .

  REFRESH gt_hist.

  DATA: lt_header TYPE TABLE OF zte3fi0002,
        ls_header TYPE zte3fi0002,
        lt_item   TYPE TABLE OF zte3fi0003,
        ls_item   TYPE zte3fi0003.

  " COLLECT 집계용 구조체
  DATA: BEGIN OF ls_sum,
          belnr TYPE zte3fi0002-belnr,
          dmbtr TYPE zte3fi0003-dmbtr,
        END OF ls_sum.
  DATA lt_sum LIKE TABLE OF ls_sum.

  DATA lv_from TYPE sy-datum.

  " 12개월 전 날짜 계산
  CALL FUNCTION 'RP_CALC_DATE_IN_INTERVAL'
    EXPORTING
      date      = sy-datum
      days      = 0
      months    = 12
      years     = 0
      signum    = '-'
    IMPORTING
      calc_date = lv_from.

  " 외화환산평가 전표 헤더 조회
  SELECT *
    INTO TABLE lt_header
    FROM zte3fi0002
   WHERE bukrs = gv_bukrs
     AND bltxt LIKE '%외화환산평가%'
     AND budat > lv_from.

  CHECK lt_header IS NOT INITIAL.
  SORT lt_header BY belnr.

  " 차변 아이템 전체 조회
  SELECT *
    INTO TABLE lt_item
    FROM zte3fi0003
    FOR ALL ENTRIES IN lt_header
   WHERE bukrs = lt_header-bukrs
     AND gjahr = lt_header-gjahr
     AND belnr = lt_header-belnr
     AND shkzg = 'S'.

  " COLLECT로 전표번호 기준 금액 합산
  LOOP AT lt_item INTO ls_item.
    CLEAR ls_sum.
    ls_sum-belnr = ls_item-belnr.
    ls_sum-dmbtr = ls_item-dmbtr.
    COLLECT ls_sum INTO lt_sum.
  ENDLOOP.

  SORT lt_sum BY belnr.

  " 헤더 + 합계 매핑
  LOOP AT lt_header INTO ls_header.
    CLEAR gs_hist.
    gs_hist-budat = ls_header-budat.
    gs_hist-bltxt = ls_header-bltxt.
    gs_hist-belnr = ls_header-belnr.
    gs_hist-waers = 'KRW'.

    READ TABLE lt_sum INTO ls_sum
      WITH KEY belnr = ls_header-belnr BINARY SEARCH.
    IF sy-subrc = 0.
      gs_hist-dmbtr = ls_sum-dmbtr.
    ENDIF.

    APPEND gs_hist TO gt_hist.
  ENDLOOP.

  " 최신순 정렬
  SORT gt_hist BY budat DESCENDING.

ENDFORM.

*& Form post_save
*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM post_save .

  DATA : lv_error(1),
         lv_answer(1).

  " 테스트 모드에서는 실제 전기 불가
  IF gv_check EQ 'X'.
    MESSAGE s000 WITH '테스트 화면에서는 전표 생성이 불가합니다.'.
    RETURN.
  ENDIF.

  IF gv_evdat NE sy-datum.
    MESSAGE s000 WITH | 전표는 { gv_evdat+4(2) }월 { gv_evdat+6(2) }일에 생성할 수 있습니다. | DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

  PERFORM check_post USING lv_error.

  IF lv_error EQ 'X'.
    MESSAGE s000 WITH |{ gv_evdat(4) }년 { gv_evdat+4(2) }월 데이터는 이미 생성되었습니다.| DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.


  DATA(lv_text) = | { gv_evdat+4(2) }월 평가/환입 전표를 생성하시겠습니까? |.
  _popup_to_confirm '평가 전표' lv_text lv_answer.

  IF lv_answer EQ '1'.

                                po_line      TYPE REF TO data
                                po_event     TYPE REF TO cl_alv_event_data.

  FIELD-SYMBOLS: <fs_line> LIKE LINE OF gt_data_disp,
                 <fs_text> TYPE any.

  DATA: lv_partner TYPE zte3sd0001-kunnr,
        lv_name    TYPE char40,
        lv_text    TYPE lvc_value.

  DATA: lv_kunnr TYPE zte3sd0001-kunnr,
        lv_lifnr TYPE zte3mm0002-lifnr,
        lv_arrow TYPE string.

  IF po_line IS NOT BOUND OR po_event IS NOT BOUND.
    EXIT.
  ENDIF.

  ASSIGN po_line->* TO <fs_line>.
  IF sy-subrc NE 0.
    EXIT.
  ENDIF.

  CASE ps_info-criteria.

    WHEN 'PARTNER'.
      " gv_expand 기준으로 화살표 결정
      lv_arrow = COND #( WHEN gv_expand = 'X'
                         THEN '▶'
                         ELSE '▼' ).

      lv_kunnr = <fs_line>-partner.
      SELECT SINGLE name1 FROM zte3sd0001
        WHERE kunnr EQ @lv_kunnr
        INTO @DATA(lv_name1).

      IF sy-subrc EQ 0 AND lv_name1 IS NOT INITIAL.
        lv_text = |{ lv_arrow } 고객: { lv_kunnr } { lv_name1 }|.

  ENDIF.

  IF sy-subrc EQ 0.

    po_event->m_event_handled = 'X'.
  ENDIF.
