FORM get_init_data .

  DATA(lv_gdatu) = 99999999 - pa_evdat.

  " 계정 마스터
  SELECT saknr stext
    INTO TABLE gt_saknr
    FROM zte3fi0001
   WHERE bukrs = gv_bukrs.
  SORT gt_saknr BY saknr.

  " 고객 마스터
  SELECT kunnr name1
    FROM zte3sd0001
    INTO TABLE gt_kunnr.
  SORT gt_kunnr BY kunnr.

  " 공급업체 마스터
  SELECT lifnr vennm
    FROM zte3mm0002
    INTO TABLE gt_lifnr.
  SORT gt_lifnr BY lifnr.

  " 평가기준일 유효 환율
  SELECT DISTINCT fcurr gdatu ukurs ffact erdat
    INTO TABLE gt_rate
    FROM zte3fi0008
   WHERE tcurr = 'KRW'
     AND fcurr IN ( 'USD', 'EUR', 'JPY' )
     AND gdatu >= lv_gdatu.

  SORT gt_rate BY fcurr gdatu ASCENDING.
  DELETE ADJACENT DUPLICATES FROM gt_rate COMPARING fcurr.

ENDFORM.

FORM get_before_data .

  DATA: ls_data      LIKE LINE OF gt_data,
        ls_rate      LIKE LINE OF gt_rate,
        lv_base_eval TYPE p DECIMALS 2.

  CLEAR gv_before.

  LOOP AT gt_data INTO ls_data WHERE belnr IS NOT INITIAL.

    READ TABLE gt_rate INTO ls_rate
      WITH KEY fcurr = ls_data-fwaer BINARY SEARCH.

    IF sy-subrc EQ 0 AND ls_data-ukurs > 0.
      lv_base_eval = ls_data-dmbtr * ls_rate-ukurs / ls_data-ukurs.
    ELSE.
      lv_base_eval = ls_data-dmbtr.
    ENDIF.

    " AR/AP별 평가손익 누적
    IF ls_data-ar_ap = 'AR'.
      gv_before = gv_before + lv_base_eval - ls_data-dmbtr.
    ELSEIF ls_data-ar_ap = 'AP'.
      gv_before = gv_before + ls_data-dmbtr - lv_base_eval.
    ENDIF.

  ENDLOOP.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form build_eval
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> GV_CHECK
*&---------------------------------------------------------------------*
FORM build_eval  USING pv_check.

  DATA: ls_header     TYPE zte3fi0002,
        lt_items      TYPE TABLE OF zte3fi0003,
        ls_item       TYPE zte3fi0003,
        lv_belnr      TYPE zte3fi0002-belnr,
        ls_rev_header TYPE zte3fi0002,
        lt_rev_items  TYPE TABLE OF zte3fi0003,
        lv_rev_belnr  TYPE zte3fi0002-belnr,
        lv_rev_date   TYPE sy-datum,
        lv_base       TYPE p DECIMALS 2.

  CHECK gv_before NE 0.

  lv_base = gv_before.

  " 평가 전표 헤더
  CLEAR ls_header.
  ls_header-gjahr = pa_evdat(4).
  ls_header-blart = 'SA'.
  ls_header-budat = pa_evdat.
  ls_header-bldat = pa_evdat.
  ls_header-monat = pa_evdat+4(2).
  ls_header-waers = 'KRW'.
  ls_header-bltxt = |{ pa_evdat(4) }년 { pa_evdat+4(2) }월 외화환산평가|.

  " 외상매입금
  CLEAR ls_item.
  ls_item-buzei = 1.
  ls_item-saknr = '2100000001'.
  ls_item-koart = 'K'.
  ls_item-waers = 'KRW'.
  ls_item-dmbtr = abs( lv_base ).
  ls_item-shkzg = COND #( WHEN lv_base < 0 THEN 'H' ELSE 'S' ).
  APPEND ls_item TO lt_items.

  " 손익계정
  CLEAR ls_item.
  ls_item-buzei = 2.
  ls_item-saknr = COND #( WHEN lv_base < 0 THEN '5300000003' ELSE '4200000003' ).
  ls_item-koart = 'S'.
  ls_item-waers = 'KRW'.
  ls_item-dmbtr = abs( lv_base ).
  ls_item-kostl = 'CC_FIN'.
  ls_item-shkzg = COND #( WHEN lv_base < 0 THEN 'S' ELSE 'H' ).
  APPEND ls_item TO lt_items.

  CHECK pv_check NE 'X'.

  " 평가 전표 생성
  zcl_e3_fi_document=>post_document(
    EXPORTING is_header = ls_header
              it_items  = lt_items
    IMPORTING ev_belnr  = lv_belnr ).

  IF lv_belnr IS INITIAL.
    WRITE : '평가전표 생성 실패.'.
    RETURN.
  ENDIF.

  WRITE: |평가전표 생성: { lv_belnr }|.

  " 익월 1일 환입 전표 생성
  lv_rev_date         = pa_evdat + 1.
  ls_rev_header       = ls_header.
  ls_rev_header-gjahr = lv_rev_date(4).
  ls_rev_header-budat = lv_rev_date.
  ls_rev_header-bldat = lv_rev_date.
  ls_rev_header-monat = lv_rev_date+4(2).
  ls_rev_header-bltxt = |{ pa_evdat(4) }년 { pa_evdat+4(2) }월 외화환산환입|.
  ls_rev_header-xblnr = lv_belnr.

  " 환입 = 평가 반대 분개
  lt_rev_items = lt_items.

  LOOP AT lt_rev_items ASSIGNING FIELD-SYMBOL(<fs_rev>).

    <fs_rev>-shkzg = COND #( WHEN <fs_rev>-shkzg = 'S' THEN 'H' ELSE 'S' ).
    IF <fs_rev>-koart = 'S'.
      <fs_rev>-saknr = COND #( WHEN <fs_rev>-saknr = '4200000003' THEN '5300000003'
                               WHEN <fs_rev>-saknr = '5300000003' THEN '4200000003'
                               ELSE <fs_rev>-saknr ).
    ENDIF.

  ENDLOOP.

  zcl_e3_fi_document=>post_document(
    EXPORTING is_header = ls_rev_header
              it_items  = lt_rev_items
    IMPORTING ev_belnr  = lv_rev_belnr ).

  IF lv_rev_belnr IS NOT INITIAL.
    WRITE: / | 환입전표 생성: { lv_rev_belnr }|.
  ELSE.
    WRITE: / | 환입전표 생성 실패 (평가전표: { lv_belnr })|.
  ENDIF.

ENDFORM.

FORM get_data .

  REFRESH gt_data.

  SELECT *
    FROM zcds_e3_fi_0001(
           p_usd = @gv_usd,
           p_eur = @gv_eur,
           p_jpy = @gv_jpy
         )
   WHERE budat <= @pa_evdat
     AND bukrs  = '1000'
     AND shkzg  = 'S'
     AND ( kunnr NE @space OR lifnr NE @space )
    INTO CORRESPONDING FIELDS OF TABLE @gt_data.

  IF gt_data IS INITIAL.
    WRITE: / '미결 외화전표가 없습니다.'.
    LEAVE PROGRAM.
  ENDIF.

  " AR/AP 구분 + 거래처명 + 평가손익 계산
  LOOP AT gt_data ASSIGNING FIELD-SYMBOL(<fs_data>).

    <fs_data>-ar_ap = SWITCH #( <fs_data>-koart
                                    WHEN 'D' THEN 'AR'
                                    WHEN 'K' THEN 'AP'
                                    ELSE          'GL' ).

    IF <fs_data>-kunnr IS NOT INITIAL.
      <fs_data>-name1   = VALUE #( gt_kunnr[ kunnr = <fs_data>-kunnr ]-name1 OPTIONAL ).
    ELSEIF <fs_data>-lifnr IS NOT INITIAL.
      <fs_data>-vennm   = VALUE #( gt_lifnr[ lifnr = <fs_data>-lifnr ]-vennm OPTIONAL ).
    ENDIF.

    PERFORM set_rate_data CHANGING <fs_data>.

  ENDLOOP.

ENDFORM.

FORM set_rate_data  CHANGING ps_data LIKE gs_data.

  DATA: lv_eval_rate TYPE p DECIMALS 9.

  READ TABLE gt_rate INTO DATA(ls_rate)
    WITH KEY fcurr = ps_data-fwaer BINARY SEARCH.

  IF sy-subrc = 0.
    lv_eval_rate = ls_rate-ukurs * ps_data-simfactor.

    IF ps_data-ukurs > 0.
      ps_data-eval_krw = ps_data-dmbtr * lv_eval_rate / ps_data-ukurs.
    ELSE.
      ps_data-eval_krw = ps_data-dmbtr.
    ENDIF.
  ENDIF.

  " AR: 평가금액 - 장부 / AP: 장부 - 평가금액
  IF ps_data-ar_ap = 'AP'.
    ps_data-eval_diff = ps_data-dmbtr - ps_data-eval_krw.
  ELSE.
    ps_data-eval_diff = ps_data-eval_krw - ps_data-dmbtr.
  ENDIF.

ENDFORM.

FORM check_post USING pv_error TYPE c.

  DATA lv_cnt TYPE i.

  SELECT COUNT(*)
    FROM zte3fi0002 INTO lv_cnt
   WHERE monat = gv_evdat+4(2)
     AND blart = 'SA'
     AND gjahr = gv_evdat(4)
     AND budat = gv_evdat
     AND bltxt LIKE '%외화환산평가%'.

  IF lv_cnt > 0.
    pv_error = 'X'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& Form sum_eval_diff
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM sum_eval_diff .

  DATA: ls_rate      LIKE LINE OF gt_rate,
        lv_base_eval TYPE p DECIMALS 2.

  LOOP AT gt_data INTO DATA(ls_data).
    READ TABLE gt_rate INTO ls_rate WITH KEY fcurr = ls_data-fwaer BINARY SEARCH.

    lv_base_eval = COND #( WHEN sy-subrc = 0 AND ls_data-ukurs > 0
                           THEN ls_data-dmbtr * ls_rate-ukurs / ls_data-ukurs
                           ELSE ls_data-dmbtr ).

    gv_before += COND #( WHEN ls_data-ar_ap = 'AR' THEN lv_base_eval - ls_data-dmbtr
                         WHEN ls_data-ar_ap = 'AP' THEN ls_data-dmbtr - lv_base_eval
                         ELSE 0 ).
  ENDLOOP.

  IF gv_before = 0.
    WRITE: / '평가손익이 0입니다. 전표 생성 대상 없음.'.
    LEAVE PROGRAM.
  ENDIF.

ENDFORM.
