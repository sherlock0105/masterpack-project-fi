*  IF gt_data IS NOT INITIAL.
*    " 3-1. 엑셀에 있는 검수번호 중 이미 완료('3')된 번호만 한 번에 DB에서 조회

*    IF sy-subrc = 0 AND lt_completed IS NOT INITIAL.

*  IF gt_data IS INITIAL.

  DATA: lt_excel     TYPE TABLE OF alsmex_tabline,

        lv_skip_cnt  TYPE i,

    EXPORTING
      filename    = gv_fname
      i_begin_col = 1
      i_begin_row = lv_begin_row
      i_end_col   = 7

    EXCEPTIONS

  CLEAR : gv_fname.
  IF sy-subrc NE 0 OR lt_excel IS INITIAL.
    MESSAGE s000 WITH '데이터가 없거나 파일을 읽을 수 없습니다.' DISPLAY LIKE 'E'. RETURN.
  ENDIF.

  "헤더검사
  SORT lt_excel BY row col.

  READ TABLE lt_excel INTO ls_hdr WITH KEY row = 1 col = 1.
  IF ls_hdr-value <> '검수번호'.
    MESSAGE s000 WITH '검수결과 업로드 양식이 아닙니다. 지정된 양식을 사용하세요.' DISPLAY LIKE 'E'. RETURN.
  ENDIF.

  READ TABLE lt_excel INTO ls_hdr WITH KEY row = 1 col = 2.
  IF ls_hdr-value <> '자재코드'.
    MESSAGE s000 WITH '검수결과 업로드 양식이 아닙니다. 지정된 양식을 사용하세요.' DISPLAY LIKE 'E'. RETURN.
  ENDIF.

  " 2. 데이터 파싱 루프
  LOOP AT lt_excel INTO ls_excel.

    " 1행 헤더 타이틀은 스킵
    IF ls_excel-row = 1.
      CONTINUE.
    ENDIF.
    IF lv_row <> 0 AND lv_row <> ls_excel-row.
      IF gs_data-prueflos IS NOT INITIAL.

        " 마스터 테이블 존재 여부 검사
        SELECT SINGLE matnr INTO @DATA(lv_matnr_chk)
          FROM zte3qm0002
         WHERE prueflos = @gs_data-prueflos.

        IF sy-subrc <> 0.
          MESSAGE s000 WITH |행({ lv_row }): 존재하지 않는 검사 ({ gs_data-prueflos }) 번호입니다.| DISPLAY LIKE 'E'.
          RETURN.
        ENDIF.

        IF lv_matnr_chk <> gs_data-matnr.
          MESSAGE s000 WITH |행({ lv_row }): { gs_data-prueflos }의 자재정보가 일치하지 않습니다.| DISPLAY LIKE 'E'.
          RETURN.
        ENDIF.

        " 에러가 없으면 무조건 추가
        APPEND gs_data TO gt_data.
      ENDIF.
      CLEAR gs_data.
    ENDIF.

    lv_row = ls_excel-row.
    CONDENSE ls_excel-value.

    CASE ls_excel-col.
      WHEN 1. gs_data-prueflos = |{ ls_excel-value ALPHA = IN }|.
      WHEN 2. gs_data-matnr    = |{ ls_excel-value ALPHA = IN }|.
      WHEN 3. gs_data-insp_cd  = ls_excel-value.
      WHEN 4. gs_data-insp_nm  = ls_excel-value.
      WHEN 5. " 수량: 음수 체크 및 숫자 체크
        " 1) 음수(-) 기호가 포함되어 있는지 확인
        IF ls_excel-value CA '-'.
          MESSAGE s000 WITH |행({ lv_row }): 폐기 수량에 음수({ ls_excel-value })를 입력할 수 없습니다.| DISPLAY LIKE 'E'.
          RETURN.

          " 2) 정상적인 숫자/소수점인지 확인
        ELSEIF ls_excel-value CO '0123456789. '.
          gs_data-fail_qty = ls_excel-value.
        ELSE.
          MESSAGE s000 WITH |행({ lv_row }): 수량에 올바른 숫자를 입력하세요.| DISPLAY LIKE 'E'.
          RETURN.
        ENDIF.
      WHEN 6. gs_data-meins    = ls_excel-value.
      WHEN 7. " 사번: 숫자 체크
        IF ls_excel-value CO '0123456789 '.
          gs_data-pernr = |{ ls_excel-value ALPHA = IN }|.

          " 1. 해당 검수번호(로트)의 검사유형 조회
          SELECT SINGLE insp_type
            INTO @DATA(lv_up_insp_type)
            FROM zte3qm0001
           WHERE prueflos = @gs_data-prueflos.
          " 2. 검사 유형에 따른 부서(BTRTL) 결정
          DATA: lv_up_team TYPE zte3cm0005-btrtl.
          IF lv_up_insp_type = '02'.
            lv_up_team = 'QM_QA'. " ★ 반품 부서
          ELSE.
            lv_up_team = 'QM_QC'. " ★ 일반 검수 부서
          ENDIF.

          SELECT SINGLE pernr
            INTO @DATA(lv_up_chk)
            FROM zte3cm0005
           WHERE pernr = @gs_data-pernr
             AND btrtl = @lv_up_team.

          IF sy-subrc NE 0.
            MESSAGE s000 WITH |행({ lv_row }): 잘못된 사번({ ls_excel-value })입니다.| DISPLAY LIKE 'E'.
            RETURN.
          ENDIF.

        ELSE.
          MESSAGE s000 WITH |행({ lv_row }): 사번에 올바른 숫자를 입력하세요.| DISPLAY LIKE 'E'.
          RETURN.
        ENDIF.
    ENDCASE.

  ENDLOOP.

  " 마지막 행 처리
  IF gs_data-prueflos IS NOT INITIAL.

    SELECT SINGLE matnr INTO @DATA(lv_last_chk)
      FROM zte3qm0002
     WHERE prueflos = @gs_data-prueflos.

    IF sy-subrc <> 0.
      MESSAGE s000 WITH |마지막 행({ lv_row }): 존재하지 않는 검사 번호입니다.| DISPLAY LIKE 'E'.
      RETURN.
    ENDIF.

    IF lv_last_chk <> gs_data-matnr.
      MESSAGE s000 WITH |마지막 행({ lv_row }): 검사로트 정보 오류| DISPLAY LIKE 'E'.
      RETURN.
    ENDIF.

    APPEND gs_data TO gt_data.
  ENDIF.

  " 3. 키 값 기준 중복 제거 및 완료 건(stats = '3') 필터링
  SORT gt_data BY prueflos insp_cd.
  DELETE ADJACENT DUPLICATES FROM gt_data COMPARING prueflos insp_cd.

  IF gt_data IS NOT INITIAL.
    " 3-1. 엑셀에 있는 검수번호 중 이미 완료('3')된 번호만 한 번에 DB에서 조회
    SELECT prueflos
      INTO TABLE @DATA(lt_completed)
      FROM zte3qm0001
       FOR ALL ENTRIES IN @gt_data
     WHERE prueflos = @gt_data-prueflos
       AND stats = '3'.

    " 3-2. 완료된 건이 존재하면 gt_data에서 안전하게 일괄 삭제
    IF sy-subrc = 0 AND lt_completed IS NOT INITIAL.
      SORT lt_completed BY prueflos.
      DELETE ADJACENT DUPLICATES FROM lt_completed COMPARING prueflos.
      lv_skip_cnt = lines( lt_completed ).

      LOOP AT lt_completed INTO DATA(ls_comp).
        DELETE gt_data WHERE prueflos = ls_comp-prueflos.
      ENDLOOP.
    ENDIF.
  ENDIF.

  " 3-3. 방어 로직: 남은 데이터가 없다면 빈 팝업을 띄우지 않고 종료
  IF gt_data IS INITIAL.
    MESSAGE s000 WITH |업로드한 파일의 모든 항목이 이미 검수 완료되었습니다.| DISPLAY LIKE 'I'.
    RETURN.
  ENDIF.

  " 1) 엑셀 상세 아이템(gt_data)에서 고유한 검수번호(PRUEFLOS) 목록만 추출
  DATA(lt_tmp_lot) = gt_data.
  SORT lt_tmp_lot BY prueflos.
  DELETE ADJACENT DUPLICATES FROM lt_tmp_lot COMPARING prueflos.

  " 2) DB에서 기본 정보(총수량, 자재코드 등)만 가져오기
  IF lt_tmp_lot IS NOT INITIAL.
    SELECT a~prueflos, a~insp_type, a~aufnr, a~budat, a~stats,
           b~matnr,    b~charg,     b~qi_qty,  b~meins,
           c~gltrp
      INTO CORRESPONDING FIELDS OF TABLE @gt_upload_pop
      FROM zte3qm0001 AS a
     INNER JOIN zte3qm0002 AS b ON a~prueflos = b~prueflos
      LEFT OUTER JOIN zte3pp0005 AS c ON a~aufnr = c~aufnr
       FOR ALL ENTRIES IN @lt_tmp_lot
     WHERE a~prueflos = @lt_tmp_lot-prueflos
       AND b~pruef_pos = '0001'.
  ENDIF.

  " 3) 엑셀에서 입력받은 항목(gt_data)의 폐기 수량을 합산하여 팝업에 반영
  LOOP AT gt_upload_pop ASSIGNING FIELD-SYMBOL(<fs_pop>).
    <fs_pop>-mark = 'X'. " ★ 2. 팝업 진입 시 기본으로 전체 선택

    " 폐기 수량 누적 변수 초기화
    <fs_pop>-fail_qty = 0.

    " 해당 검수번호(로트)의 엑셀 데이터(gt_data)를 순회하며 폐기 수량 누적
    LOOP AT gt_data INTO DATA(ls_tmp_data) WHERE prueflos = <fs_pop>-prueflos.
      <fs_pop>-fail_qty = <fs_pop>-fail_qty + ls_tmp_data-fail_qty.
    ENDLOOP.

    " 합격 수량 = 총수량(qi_qty) - 엑셀에 적힌 총 폐기수량(fail_qty)
    <fs_pop>-pass_qty = <fs_pop>-qi_qty - <fs_pop>-fail_qty.
  ENDLOOP.

  " ==========================================================
  " ★ 3. ALV 그리드가 이미 메모리에 떠있다면, 최신 데이터로 화면 즉시 갱신!
  " ==========================================================
  IF gcl_grid_popup IS BOUND.
    gcl_grid_popup->refresh_table_display(
      EXPORTING
        is_stable      = gs_stable
        i_soft_refresh = space ).
  ENDIF.

  " 4) 모달 팝업 호출
  CALL SCREEN '0200' STARTING AT 20 5.

  " ==========================================================
  " 저장 처리 로직
  " ==========================================================
  IF gv_okcode = 'SAVE'.

    " 체크 안 된 항목 수집
    LOOP AT gt_upload_pop INTO gs_upload_pop WHERE mark = space.
      APPEND gs_upload_pop-prueflos TO lt_del_lot.
    ENDLOOP.

    " 실제 저장할 엑셀 아이템 데이터(gt_data)에서 선택 안 된 로트를 삭제
    IF lt_del_lot IS NOT INITIAL.
      LOOP AT lt_del_lot INTO DATA(lv_del_prueflos).
        DELETE gt_data WHERE prueflos = lv_del_prueflos.
      ENDLOOP.
    ENDIF.

    " 전부 체크 해제하고 눌렀을 경우 방어
    IF gt_data IS INITIAL.
      MESSAGE s000 WITH '선택된 검수 항목이 없어 취소되었습니다.' DISPLAY LIKE 'W'.
      RETURN.
    ENDIF.

    " 최종 건수 다시 계산
    DATA(lt_popup_count) = gt_data.
    SORT lt_popup_count BY prueflos.
    DELETE ADJACENT DUPLICATES FROM lt_popup_count COMPARING prueflos.
    lv_lot_cnt = lines( lt_popup_count ).

    " 실제 DB 반영
    MOVE-CORRESPONDING gt_data TO lt_ins_table.
    MODIFY zte3qm0004 FROM TABLE lt_ins_table.

    IF sy-subrc = 0.
      PERFORM calc_update.
      PERFORM get_excel_data USING lv_skip_cnt.
      MESSAGE s000 WITH |선택한 { lv_lot_cnt }건의 검수 내역이 성공적으로 저장되었습니다.|.
    ELSE.
      ROLLBACK WORK.
      MESSAGE s000 WITH '저장 중 데이터베이스 오류 발생' DISPLAY LIKE 'E'.
    ENDIF.

  ELSE.
    " 팝업에서 취소/종료(CANC)를 누른 경우
    MESSAGE s000 WITH '엑셀 업로드가 취소되었습니다.' DISPLAY LIKE 'S'.
    RETURN.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*

*& <--  p2        text
*&---------------------------------------------------------------------*
FORM set_title .

  CLEAR: gv_total, gv_wait, gv_done, gv_rate, gv_fail_amt.

  DATA: lv_pass  TYPE p DECIMALS 3,
        lv_qi    TYPE p DECIMALS 3,
        lv_perid TYPE zte3co0014-perid,
        lv_c(10).

  REFRESH gt_cnt.

  SELECT prueflos, stats, matnr, qi_qty, budat,
         pass_qty, fail_qty, charg, meins
    FROM zcds_e3_qm_0001
    INTO CORRESPONDING FIELDS OF TABLE @gt_cnt.

  gv_total = lines( gt_cnt ).

  LOOP AT gt_cnt INTO gs_cnt.

    CASE gs_cnt-stats.
      WHEN '1'. " 검수대기
        gv_wait = gv_wait + 1.

      WHEN '3'. " 결과조회(완료)
        gv_done = gv_done + 1.
        lv_pass = lv_pass + gs_cnt-pass_qty.
        lv_qi   = lv_qi   + gs_cnt-qi_qty.

        " 폐기금액 계산 — 표준원가(gt_price)에서 월별 단가 매핑
        IF gs_cnt-fail_qty > 0.
          lv_perid = gs_cnt-budat+4(2). " 전기일자(budat)에서 '월' 2자리 추출

          READ TABLE gt_price INTO DATA(ls_price)
            WITH KEY matnr = gs_cnt-matnr
                     perid = lv_perid.

          IF sy-subrc EQ 0.
            " 폐기수량 * 표준원가 총액
            gv_fail_amt += gs_cnt-fail_qty * ls_price-total.
          ENDIF.
        ENDIF.
    ENDCASE.

  ENDLOOP.

  " 합격률 및 폐기율 계산
  IF lv_qi > 0.
    gv_rate = ( lv_pass / lv_qi ) * 100.
  ENDIF.
  gv_fail_rate = 100 - gv_rate.

  " UI 화면 출력용 문자열 포맷팅 및 공백 제거 (CONDENSE)
  lv_c = gv_total.
  CONDENSE lv_c.
  WRITE lv_c TO gv_total.
  lv_c = gv_done.
  CONDENSE lv_c.
  WRITE lv_c TO gv_done.
  lv_c = gv_wait.
  CONDENSE lv_c.
  WRITE lv_c TO gv_wait.

  " 6. 대시보드 화면 갱신을 위한 커맨드 강제 유발
  cl_gui_cfw=>set_new_ok_code( new_code = 'A' ).

ENDFORM.

*&---------------------------------------------------------------------*
*& Form calc_update
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM calc_update .

  DATA: lv_total_fail TYPE zte3qm0004-fail_qty,
        lv_total_pass TYPE p DECIMALS 3,
        lv_qi_qty     TYPE zte3qm0002-qi_qty,
        lv_charg      TYPE zte3qm0002-charg,
        lv_upd_pernr  TYPE zte3cm0005-pernr,
        lv_subrc      TYPE sy-subrc.

  DATA: ls_price_key  LIKE LINE OF gt_price.

  SORT gt_data BY prueflos.

  LOOP AT gt_data ASSIGNING FIELD-SYMBOL(<fs_data>).

    AT NEW prueflos.
      CLEAR: lv_total_fail, lv_qi_qty.
    ENDAT.

    lv_total_fail = lv_total_fail + <fs_data>-fail_qty.

    AT END OF prueflos.
      " 1. 헤더 정보 조회
      SELECT SINGLE qi_qty, charg
        FROM zte3qm0002
        INTO (@lv_qi_qty, @lv_charg)
       WHERE prueflos = @<fs_data>-prueflos.


      lv_total_pass = lv_qi_qty - lv_total_fail.

      " 담당자 결정
      lv_upd_pernr = COND #( WHEN <fs_data>-pernr IS NOT INITIAL
                             THEN <fs_data>-pernr
                             ELSE gv_insp_pernr ).

      " 2. DB 업데이트
      PERFORM update_process USING <fs_data> lv_qi_qty lv_charg
                                    lv_total_pass lv_total_fail lv_upd_pernr.

      " 3. 폐기 자재 전표 생성
      IF lv_total_fail > 0.
        READ TABLE gt_price INTO ls_price_key
             WITH KEY matnr = <fs_data>-matnr.

        IF sy-subrc EQ 0.
          CLEAR lv_subrc.
          PERFORM post_fi_document USING    <fs_data>
                                           lv_total_fail
                                           ls_price_key
                                  CHANGING lv_subrc.

          IF lv_subrc = 0.
            PERFORM post_mm_document USING    <fs_data>
                                              lv_charg
                                              lv_total_fail
                                              ls_price_key
                                     CHANGING lv_subrc.
          ENDIF.

          " 오류 발생 시 롤백 및 종료
          IF lv_subrc NE 0.
            ROLLBACK WORK.
            MESSAGE '전표 생성 중 오류가 발생하여 취소되었습니다.' TYPE 'S' DISPLAY LIKE 'E'.
            RETURN.
          ENDIF.
        ENDIF.
      ENDIF.

    ENDAT.
  ENDLOOP.

  COMMIT WORK AND WAIT.
  CLEAR gv_insp_pernr.

ENDFORM.

FORM save_qi_data  USING pt_query TYPE cnht_query_table.

  DATA: ls_hdr1    TYPE zte3qm0001,
        ls_hdr2    TYPE zte3qm0002,
        ls_data    TYPE zte3qm0004,
        lt_data    TYPE TABLE OF zte3qm0004,
        lv_qty     TYPE zte3qm0004-fail_qty,
        lv_insp_nm TYPE zte3cm0005-pernam,
        lv_pernr   TYPE zte3cm0005-pernr.

  TYPES: BEGIN OF ty_insp,
           key     TYPE string,
           insp_cd TYPE string,
           insp_nm TYPE string,
         END OF ty_insp.

  DATA: lt_insp TYPE TABLE OF ty_insp,
        ls_insp TYPE ty_insp.

  DATA lv_answer(1).

  lt_insp = VALUE #(
    ( key = 'Q1'  insp_cd = '1'  insp_nm = '색상' )
    ( key = 'Q2'  insp_cd = '2'  insp_nm = '이물질' )
    ( key = 'Q10' insp_cd = '10' insp_nm = 'PH' )
    ( key = 'Q11' insp_cd = '11' insp_nm = '용량' )
    ( key = 'Q12' insp_cd = '12' insp_nm = '납/수은' )
    ( key = 'Q20' insp_cd = '20' insp_nm = '방부제' )
    ( key = 'Q21' insp_cd = '21' insp_nm = '밀봉/상태' )
    ( key = 'Q31' insp_cd = '31' insp_nm = '라벨/LOT' ) ).


  LOOP AT pt_query ASSIGNING FIELD-SYMBOL(<ls_q>).
    IF <ls_q>-name EQ 'PERNR'.
      lv_pernr = <ls_q>-value.
    ENDIF.
  ENDLOOP.

  IF lv_pernr IS INITIAL.
    MESSAGE s000 WITH '검사자 사번을 입력하세요.' DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

  SELECT SINGLE prueflos, stats, insp_type
    INTO CORRESPONDING FIELDS OF @ls_hdr1
    FROM zte3qm0001
   WHERE prueflos = @gv_save_prueflos.

  DATA: lv_team TYPE zte3cm0005-btrtl.

  CASE ls_hdr1-insp_type.
    WHEN '02'.         " 반품 입고 검사
      lv_team = 'QM_RT'.
    WHEN OTHERS.       " 그 외 일반 검사 (01, 03 등)
      lv_team = 'QM_QC'.
  ENDCASE.

  " 3. 사번 존재 여부 및 분기된 부서(lv_team) 일치 여부 체크
  SELECT SINGLE pernr FROM zte3cm0005
    INTO @DATA(lv_chk)
   WHERE pernr EQ @lv_pernr
     AND btrtl EQ @lv_team.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH |해당 부서({ lv_team })에 소속된 검사자가 아닙니다: { lv_pernr }| DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

  SELECT SINGLE prueflos, stats
    INTO CORRESPONDING FIELDS OF @ls_hdr1
    FROM zte3qm0001
   WHERE prueflos = @gv_save_prueflos.

  SELECT SINGLE prueflos, matnr, qi_qty, meins
    INTO CORRESPONDING FIELDS OF @ls_hdr2
    FROM zte3qm0002
   WHERE prueflos = @gv_save_prueflos.

  LOOP AT pt_query ASSIGNING <ls_q>.
    READ TABLE lt_insp INTO ls_insp WITH KEY key = <ls_q>-name.
    IF sy-subrc EQ 0.
      lv_qty = <ls_q>-value.
      CLEAR ls_data.
      ls_data-prueflos = gv_save_prueflos.
      ls_data-matnr    = ls_hdr2-matnr.
      ls_data-insp_cd  = ls_insp-insp_cd.
      ls_data-insp_nm  = ls_insp-insp_nm.
      ls_data-fail_qty = lv_qty.
      ls_data-meins    = ls_hdr2-meins.
      APPEND ls_data TO lt_data.
    ENDIF.
  ENDLOOP.

  " 불량합계 > 총수량 체크
  DATA lv_total_fail TYPE p DECIMALS 3.
  LOOP AT lt_data INTO ls_data.
    lv_total_fail += ls_data-fail_qty.
  ENDLOOP.

  IF lv_total_fail > ls_hdr2-qi_qty.
    MESSAGE s000 WITH '불량수량 합계가 총수량을 초과합니다.' DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

  CALL FUNCTION 'POPUP_TO_CONFIRM'
    EXPORTING
      titlebar              = '검사 결과 저장'
      text_question         = |{ gv_save_prueflos } 검사 결과를 저장하시겠습니까?|
      text_button_1         = '예'
      icon_button_1         = '@01@'
      text_button_2         = '아니오'
      icon_button_2         = '@0W@'
      default_button        = '1'
      display_cancel_button = ' '
    IMPORTING
      answer                = lv_answer.

  IF lv_answer NE '1'. RETURN. ENDIF.

  IF lt_data IS NOT INITIAL.
    MODIFY zte3qm0004 FROM TABLE lt_data.
  ENDIF.

  IF sy-subrc EQ 0.
    COMMIT WORK AND WAIT.

    gv_insp_pernr = lv_pernr.

    DELETE gt_data WHERE prueflos = gv_save_prueflos.

    SELECT prueflos matnr insp_cd insp_nm fail_qty meins
      APPENDING CORRESPONDING FIELDS OF TABLE gt_data
       FROM zte3qm0004
      WHERE prueflos = gv_save_prueflos
      ORDER BY insp_cd ASCENDING.

    PERFORM calc_update.
    PERFORM show_html USING gv_save_prueflos.
    PERFORM set_title.
    MESSAGE '검사 결과가 저장되었습니다.' TYPE 'S'.

  ELSE.
    ROLLBACK WORK.
    MESSAGE '저장 실패.' TYPE 'S' DISPLAY LIKE 'E'.
  ENDIF.

ENDFORM.

FORM get_header_data USING pv_filter.

  REFRESH gt_header.
  " 1. 상태값 필터 조건 변환
  DATA: lr_stats     TYPE RANGE OF zte3qm0001-stats,
        lr_insp_type TYPE RANGE OF zte3qm0001-insp_type.

  IF pv_filter NE 'ALL'.
    lr_stats = VALUE #( ( low = pv_filter sign = 'I' option = 'EQ' ) ).
  ENDIF.

  " 검수유형 조건 - AL이면 전체
  IF gv_insp_type NE 'AL' AND gv_insp_type IS NOT INITIAL.
    lr_insp_type = VALUE #( ( low = gv_insp_type sign = 'I' option = 'EQ' ) ).
  ELSE.
    " 전체: 존재하는 유형 다 담기
    SELECT DISTINCT domvalue_l
      INTO TABLE @DATA(lt_types)
      FROM dd07t
     WHERE domname = 'ZDE3_QM_INSP_TYPE'.

    lr_insp_type = VALUE #( FOR ls_t IN lt_types
                            ( low = ls_t-domvalue_l sign = 'I' option = 'EQ' ) ).
  ENDIF.

  SELECT prueflos,  insp_type,  ebeln,   aufnr,   budat,
         stats,     pernr,     matnr,   charg,   qi_qty,
         pass_qty,  fail_qty,   meins,   gltrp,   icon,
         stats_txt
    FROM zcds_e3_qm_0001
   WHERE stats IN @lr_stats
         AND insp_type IN @lr_insp_type
     AND ( @gv_aufnr IS INITIAL OR aufnr = @gv_aufnr )
     AND ( @gv_matnr IS INITIAL OR matnr = @gv_matnr )
     AND ( @gv_charg IS INITIAL OR charg = @gv_charg )
     AND ( ( @gv_min_date IS INITIAL AND @gv_max_date IS INITIAL )
      OR gltrp BETWEEN @gv_min_date AND @gv_max_date )
   ORDER BY stats ASCENDING, budat DESCENDING, gltrp DESCENDING, prueflos ASCENDING
    INTO CORRESPONDING FIELDS OF TABLE @gt_header.

  " 3. 조회 결과 메시지 처리
  IF sy-subrc NE 0.
    MESSAGE s000 WITH '선택한 조건에 해당되는 데이터가 없습니다.'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& Form build_html_rows
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> LT_DETAIL
*&      <-- LV_ROWS_ALL
*&---------------------------------------------------------------------*
FORM build_html_rows    USING  it_detail TYPE table
                     CHANGING  cv_rows_all TYPE string.

  DATA: ls_detail       TYPE zte3qm0004,
        lv_fail_qty(20) TYPE c,
        lv_tag          TYPE string,
        lv_row          TYPE string.

  " --- HTML 행 정렬을 위한 임시 테이블 선언 ---
  TYPES: BEGIN OF ty_html_sort,
           insp_cd_num TYPE i,            " 숫자로 정렬하기 위한 Integer 필드
           detail      TYPE zte3qm0004,
         END OF ty_html_sort.

  DATA: lt_html_sort TYPE TABLE OF ty_html_sort,
        ls_html_sort TYPE ty_html_sort.

  " 1. 들어온 데이터를 임시 테이블에 넣으면서 검사코드를 숫자로 자동 변환
  LOOP AT it_detail INTO ls_detail.
    CLEAR ls_html_sort.
    ls_html_sort-insp_cd_num = ls_detail-insp_cd. " 알파벳/공백 제외한 숫자가 캐스팅됨
    ls_html_sort-detail      = ls_detail.
    APPEND ls_html_sort TO lt_html_sort.
  ENDLOOP.

  " 2. 텍스트 기준이 아닌 숫자 기준으로 오름차순 정렬 (1, 2, 10, 11...)
  SORT lt_html_sort BY insp_cd_num ASCENDING.

  " 3. 정렬된 임시 테이블을 가지고 HTML <tr> 태그 생성
  LOOP AT lt_html_sort INTO ls_html_sort.
    ls_detail = ls_html_sort-detail.

    WRITE ls_detail-fail_qty TO lv_fail_qty DECIMALS 0.

    IF ls_detail-fail_qty > 0.
      lv_tag = |<span style="color:#FF0000;font-weight:600">폐기</span>|.
    ELSE.
      lv_tag = |<span style="color:#0a7f2e;font-weight:600">합격</span>|.
    ENDIF.

    lv_row = |<tr><td>{ ls_detail-insp_cd }</td>|
          && |<td>{ ls_detail-insp_nm }</td>|
          && |<td class="num">{ lv_fail_qty }</td>|
          && |<td>{ ls_detail-meins }</td>|
          && |<td>{ lv_tag }</td></tr>|.

*&---------------------------------------------------------------------*

*& <--  p2        text

FORM check_data.

  CLEAR gv_error.

  IF gv_aufnr    IS INITIAL AND
     gv_matnr    IS INITIAL AND
     gv_charg    IS INITIAL AND
     gv_min_date IS INITIAL AND
     gv_max_date IS INITIAL.

    MESSAGE s000 WITH '검색 조건을 하나 이상 입력해주세요' DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

  IF gv_aufnr IS NOT INITIAL.
    SELECT SINGLE aufnr
      FROM zte3qm0001
      INTO @DATA(lv_aufnr)
     WHERE aufnr = @gv_aufnr.

    IF sy-subrc NE 0.
      MESSAGE s000 WITH '존재하지 않는 생산번호입니다' DISPLAY LIKE 'E'.
      gv_error = 'X'.
      RETURN.
    ENDIF.
  ENDIF.

  IF gv_matnr IS NOT INITIAL.
    SELECT SINGLE matnr FROM zte3mm0001 INTO @DATA(lv_matnr)
     WHERE matnr = @gv_matnr.
    IF sy-subrc NE 0.
      MESSAGE s000 WITH '존재하지 않는 자재코드입니다' DISPLAY LIKE 'E'.
      gv_error = 'X'.
      RETURN.
    ENDIF.
  ENDIF.

  IF gv_charg IS NOT INITIAL.

    SELECT SINGLE charg
      FROM zte3qm0002
      INTO @DATA(lv_charg)
     WHERE charg = @gv_charg.

    IF sy-subrc NE 0.
      MESSAGE s000 WITH '존재하지 않는 배치번호입니다' DISPLAY LIKE 'E'.
      gv_error = 'X'.
      RETURN.
    ENDIF.
  ENDIF.

  IF gv_min_date IS NOT INITIAL AND gv_max_date IS NOT INITIAL.
    IF gv_min_date > gv_max_date.
      MESSAGE s000 WITH '시작일이 종료일보다 클 수 없습니다' DISPLAY LIKE 'E'.
      gv_error = 'X'.
      RETURN.
    ENDIF.
  ENDIF.

  IF gv_min_date IS INITIAL AND gv_max_date IS INITIAL.
    MESSAGE s000 WITH '날짜는 필수 입력입니다.' DISPLAY LIKE 'E'.
    gv_error = 'X'.
    RETURN.
  ENDIF.

  IF gv_max_date - gv_min_date > 31.
    MESSAGE s000 WITH '조회 기간은 최대 1개월(31일)까지만 가능합니다' DISPLAY LIKE 'E'.
    gv_error = 'X'.
    RETURN.
  ENDIF.

ENDFORM.

          maktx TYPE zte3mm0001-maktx,

      INNER JOIN zte3qm0002 AS b ON b~prueflos = a~prueflos
      INNER JOIN zte3mm0001 AS c ON c~matnr    = b~matnr

      INNER JOIN zte3mm0001 AS c ON c~matnr    = b~matnr
      INNER JOIN zte3pp0005 AS p ON p~aufnr    = a~aufnr
     WHERE p~gltrp BETWEEN gv_min_date AND gv_max_date.

  ENDIF.

  SORT lt_val BY aufnr.
