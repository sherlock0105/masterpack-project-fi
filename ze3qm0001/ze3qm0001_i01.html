MODULE user_command_0200 INPUT.

*  CASE gv_okcode.
*    WHEN 'SAVE'.
*      IF gcl_grid_popup IS BOUND.
*        gcl_grid_popup->check_changed_data( ).
*      ENDIF.
*    WHEN OTHERS.
*  ENDCASE.

  DATA: lv_answer  TYPE c,
        lv_chk_cnt TYPE i.

  " ALV 그리드의 변경사항을 인터널 테이블에 반영
  IF gcl_grid_popup IS BOUND.
    gcl_grid_popup->check_changed_data( ).
  ENDIF.

  CASE sy-ucomm.
    WHEN 'SAVE'.
      " 1) 체크된 항목이 있는지 검사
      CLEAR lv_chk_cnt.
      LOOP AT gt_upload_pop TRANSPORTING NO FIELDS WHERE mark = 'X'.
        lv_chk_cnt = lv_chk_cnt + 1.
      ENDLOOP.

      IF lv_chk_cnt = 0.
        MESSAGE s000 WITH '선택된 검수 항목이 없습니다.' DISPLAY LIKE 'W'.
        CLEAR sy-ucomm.
        EXIT. " 화면을 닫지 않고 빠져나감
      ENDIF.

      " 2) 0200 화면이 떠 있는 상태에서 확인 팝업 호출
      CALL FUNCTION 'POPUP_TO_CONFIRM'
        EXPORTING
          titlebar              = '검사 결과 저장'
          text_question         = |선택하신 { lv_chk_cnt }건의 검사 결과를 저장하시겠습니까?|
          text_button_1         = '예'
          icon_button_1         = '@01@'
          text_button_2         = '아니오'
          icon_button_2         = '@0W@'
          default_button        = '1'
          display_cancel_button = ' '
        IMPORTING
          answer                = lv_answer.

      " 3) 팝업 응답 처리
      IF lv_answer EQ '1'.
        " '예'를 누르면 확인 코드를 세팅하고 0200 팝업 종료
        gv_okcode = 'SAVE'.
        LEAVE TO SCREEN 0.
      ELSE.
        " '아니오'를 누르면 아무 동작 없이 0200 화면 유지
        MESSAGE s000 WITH '저장이 취소되었습니다.' DISPLAY LIKE 'S'.
        CLEAR sy-ucomm.
      ENDIF.

    WHEN 'CANC' OR 'EXIT' OR 'BACK'.
      " 0200 팝업창 닫기 버튼 클릭 시
      gv_okcode = 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.
